00291 /* 
00292  * Sort the rays (Ray, Sat) into three tiers as used in 'Chernikova' function:
00293  * NbBid         <= i <  equal_bound    : saturates the constraint
00294  * equal_bound   <= i <  sup_bound      : verifies the constraint
00295  * sup_bound     <= i <  NbRay          : does not verify 
00296  *
00297  * 'Ray' is the matrix of rays and 'Sat' is the corresponding saturation 
00298  * matrix. (jx,bx) pair specify the constraint in the saturation matrix. The
00299  * status element of the 'Ray' matrix holds the saturation value w.r.t the 
00300  * constraint specified by (jx,bx). Thus
00301  * Ray->p[i][0]  = 0 -> ray(i) saturates the constraint
00302  * Ray->p[i][0]  > 0 -> ray(i) verifies  the constraint
00303  * Ray->p[i][0]  < 0 -> ray(i) doesn't verify the constraint  
00304  */  
00305 static void RaySort(Matrix *Ray,SatMatrix *Sat,int NbBid,int NbRay,int *equal_bound,int *sup_bound,unsigned RowSize1, unsigned RowSize2, unsigned bx, unsigned jx) {                     
00306 
00307   int inf_bound;
00308   Value **uni_eq, **uni_sup, **uni_inf;
00309   int **inc_eq, **inc_sup, **inc_inf;
00310 
00311   /* 'uni_eq' points to the first ray in the ray matrix which verifies a
00312    * constraint, 'inc_eq' is the corresponding pointer in saturation 
00313    * matrix. 'uni_inf' points to the first ray (from top) which doesn't 
00314    * verify a constraint, 'inc_inf' is the corresponding pointer in 
00315    * saturation matrix. 'uni_sup' scans the ray matrix and 'inc_sup' is 
00316    * the corresponding pointer in saturation matrix. 'inf_bound' holds the 
00317    * number of the first ray which does not verify the constraints. 
00318    */
00319 
00320   *sup_bound = *equal_bound = NbBid;
00321   uni_sup = uni_eq = Ray->p+NbBid;
00322   inc_sup = inc_eq = Sat->p+NbBid;
00323   inf_bound = NbRay;
00324   uni_inf = Ray->p+NbRay;
00325   inc_inf = Sat->p+NbRay;
00326   
00327   while (inf_bound>*sup_bound) {
00328     if (value_zero_p(**uni_sup)) {               /* status = satisfy */
00329       if (inc_eq != inc_sup) {
00330         Vector_Exchange(*uni_eq,*uni_sup,RowSize1);
00331         bexchange(*inc_eq,*inc_sup,RowSize2);
00332       }
00333       (*equal_bound)++; uni_eq++; inc_eq++;
00334       (*sup_bound)++; uni_sup++; inc_sup++;
00335     }
00336     else {
00337       *((*inc_sup)+jx)|=bx;
00338       
00339       /* if (**uni_sup<0) */
00340       if (value_neg_p(**uni_sup)) {             /* Status != verify  */
00341         inf_bound--; uni_inf--; inc_inf--;
00342         if (inc_inf != inc_sup) {
00343           Vector_Exchange(*uni_inf,*uni_sup,RowSize1);
00344           bexchange(*inc_inf,*inc_sup,RowSize2);
00345         }
00346       }
00347       else {                                     /* status == verify */
00348          (*sup_bound)++; uni_sup++; inc_sup++;
00349       } 
00350     }
00351   }
00352 } /* RaySort */

00583                 if (!(aux&m)) 
00584                   nbcommonconstraints++;
00585             }
00586             aux = Temp[jx] =  Sat->p[i][jx] | Sat->p[j][jx];
00587             for (m=MSB; m!=bx; m>>=1) 
00588               if (!(aux&m)) 
00589                 nbcommonconstraints++;      
00590             rayonly = (value_zero_p(Ray->p[i][Dimension])  && 
00591                        value_zero_p(Ray->p[j][Dimension]) && 
00592                        (dual == 0));                
00593             if(rayonly)
00594               nbcommonconstraints++;      /* account for pos constr */
00595 
00596           /*-----------------------------------------------------------------*/
00597           /* Adjacency Test : is combination [R-,R+] a non redundant ray?    */
00598           /*-----------------------------------------------------------------*/
00599           
00600             if (nbcommonconstraints+NbBid>=Dimension-2) { /* Dimensionality check*/
00601               /* Check whether a ray m saturates the same set of constraints */
00602               redundant=0;
00603               for (m=NbBid; m<bound; m++) 
00604                 if ((m!=i)&&(m!=j)) {
00605                   
00606                   /* Two rays (r+ r-) are never made redundant by a vertex */
00607                   /* because the positivity constraint saturates both rays */
00608                   /* but not the vertex                                    */
00609                   
00610                   if (rayonly && value_notzero_p(Ray->p[m][Dimension]))
00611                     continue;
00612 
00613                   /* (r+ r-) is redundant if there doesn't exist an equation */
00614                   /* which saturates both r+ and r- but not rm.              */
00615                   
00616                   ip1 = Temp;
00617                   ip2 = Sat->p[m];
00618                   for (l=0; l<=jx; l++,ip2++,ip1++)
00619                     if (*ip2 & ~*ip1) 
00620                       break;
00621                   if (l>jx) { 
00622                     redundant=1;
00623                     break;
00624                   }
00625                 }
00626 
00627 #ifdef POLY_CH_DEBUG
00628               fprintf(stderr, "[Chernikova: E]\nRay = ");
00629               Ray->NbRows=NbRay;
00630               Matrix_Print(stderr,0,Ray);
00631               fprintf(stderr, "\nConstraints = ");
00632               Matrix_Print(stderr,0,Mat);
00633               fprintf(stderr, "\nSat = ");
00634               SMPrint (Sat);
00635 #endif
00636               
00637               /*------------------------------------------------------------*/
00638               /* Add new ray generated by [R+,R-]                           */
00639               /*------------------------------------------------------------*/
00640             
00641               if (!redundant) {
00642                 if (NbRay==NbMaxRays) {
00643                   NbMaxRays *= 2;
00644                   Ray->NbRows = NbRay;
00645                   Matrix_Extend(Ray, NbMaxRays);
00646                   SatMatrix_Extend(Sat, Mat, NbMaxRays);
00647                 }
00648                 
00649                 /* Compute the new ray */
00650                 Combine(Ray->p[j],Ray->p[i],Ray->p[NbRay],0,Dimension);
00651                 SatVector_OR(Sat->p[j],Sat->p[i],Sat->p[NbRay],sat_nbcolumns);
00652                 Sat->p[NbRay][jx] &= ~bx;
00653                 NbRay++;
00654               }
00655             }
00656           }
00657 
00658 #ifdef POLY_CH_DEBUG
00659         fprintf(stderr, 
00660                 "[Chernikova: F]\n"
00661                 "sup_bound=%d\n"
00662                 "equal_bound=%d\n"
00663                 "bound=%d\n"
00664                 "NbRay=%d\n"
00665                 "Dimension = %d\n"
00666                 "Ray = ",sup_bound,equal_bound,bound,NbRay,Dimension);
00667 #endif
00668 #ifdef POLY_CH_DEBUG
00669         Ray->NbRows=NbRay;
00670         fprintf(stderr, "[Chernikova: F]:\nRay = ");
00671         Matrix_Print(stderr,0,Ray);
00672 #endif
00673         
00674         /* Eliminates all non extremal rays */
00675         /* j = (Mat->p[k][0]) ? */
00676         
00677         j = (value_notzero_p(Mat->p[k][0])) ? 
00678           sup_bound : equal_bound;
00679         
00680         i = NbRay;
00681 #ifdef POLY_CH_DEBUG
00682         fprintf(stderr, "i = %d\nj = %d \n", i, j);
00683 #endif
00684         while ((j<bound)&&(i>bound)) {
00685           i--;
00686           Vector_Copy(Ray->p[i],Ray->p[j],Dimension+1);
00687           SMVector_Copy(Sat->p[i],Sat->p[j],sat_nbcolumns);
00688           j++;
00689         }
00690 
00691 #ifdef POLY_CH_DEBUG
00692         fprintf(stderr, "i = %d\nj = %d \n", i, j);
00693         fprintf(stderr, 
00694                 "[Chernikova: F]\n"
00695                 "sup_bound=%d\n"
00696                 "equal_bound=%d\n"
00697                 "bound=%d\n"
00698                 "NbRay=%d\n"
00699                 "Dimension = %d\n"
00700                 "Ray = ",sup_bound,equal_bound,bound,NbRay, Dimension);
00701 #endif
00702 #ifdef POLY_CH_DEBUG
00703         Ray->NbRows=NbRay;
00704         fprintf(stderr, "[Chernikova: G]\nRay = "); 
00705         Matrix_Print(stderr,0,Ray);
00706 #endif  
00707         if (j==bound) 
00708           NbRay=i;
00709         else 
00710           NbRay=j;
00711       }
00712       NEXT(jx,bx);
00713     }    
00714     Ray->NbRows=NbRay;
00715     Sat->NbRows=NbRay;
00716     
00717   } /* End of TRY */
00718 
00719   UNCATCH(any_exception_error);
00720   free(Temp);
00721   
00722 #ifdef POLY_CH_DEBUG
00723   fprintf(stderr, "[Chernikova: Output]\nRay = ");
00724   Matrix_Print(stderr,0,Ray);
00725   fprintf(stderr, "\nConstraints = ");
00726   Matrix_Print(stderr,0,Mat);
00727   fprintf(stderr, "\nSat = ");
00728   SMPrint (Sat);
00729 #endif
00730   
00731   return 0;
00732 } /* Chernikova */

00734 static int Gauss4(Value **p, int NbEq, int NbRows, int Dimension)
00735 {
00736   int i, j, k, pivot, Rank;
00737   int *column_index = NULL;
00738   Value gcd;
00739 
00740   value_init(gcd);
00741   column_index=(int *)malloc(Dimension * sizeof(int));
00742   if(!column_index) {   
00743     errormsg1("Gauss","outofmem","out of memory space");
00744     value_clear(gcd);
00745     return 0;
00746   }
00747   Rank=0;
00748   
00749   CATCH(any_exception_error) {
00750     if (column_index)
00751       free(column_index);
00752     value_clear(gcd);
00753     RETHROW();
00754   }
00755   TRY {
00756     
00757     for (j=1; j<=Dimension; j++) {   /* for each column (except status) */  
00758       for (i=Rank; i<NbEq; i++)      /* starting at diagonal, look down */
00759         
00760         /* if (Mat->p[i][j] != 0) */
00761         if (value_notzero_p(p[i][j])) 
00762           break;                    /* Find the first non zero element  */    
00763       if (i!=NbEq) {                /* If a non-zero element is found?  */
00764         if (i!=Rank)                /* If it is found below the diagonal*/
00765           Vector_Exchange(p[Rank]+1,p[i]+1,Dimension);
00766         
00767         /* Normalize the pivot row by dividing it by the gcd */      
00768         /* gcd = Vector_Gcd(p[Rank]+1,Dimension) */
00769         Vector_Gcd(p[Rank]+1,Dimension,&gcd);
00770         
00771         if (value_cmp_si(gcd, 2) >= 0)
00772           Vector_AntiScale(p[Rank]+1, p[Rank]+1, gcd, Dimension);
00773         
00774         if (value_neg_p(p[Rank][j]))
00775           Vector_Oppose(p[Rank]+1, p[Rank]+1, Dimension);
00776         
00777         pivot=i;
00778         for (i=pivot+1; i<NbEq; i++) {  /* Zero out the rest of the column */
00779           
00780           /* if (Mat->p[i][j] != 0) */
00781           if (value_notzero_p(p[i][j]))
00782             Combine(p[i],p[Rank],p[i],j,Dimension);
00783         }
00784         
00785         /* For each row with non-zero entry Mat->p[Rank], store the column */
00786         /* number 'j' in 'column_index[Rank]'. This information will be    */
00787         /* useful in performing Gaussian elimination backward step.        */
00788         
00789         column_index[Rank]=j;
00790         Rank++;
00791       }
00792     } /* end of Gaussian elimination forward step */
00793     
00794     /* Back Substitution -- normalize the system of equations */
00795     for (k=Rank-1; k>=0; k--) { 
00796       j = column_index[k];
00797 
00798       /* Normalize the equations */
00799       for (i=0; i<k; i++) { 
00800         
00801         /* if (Mat->p[i][j] != 0) */
00802         if (value_notzero_p(p[i][j]))
00803           Combine(p[i],p[k],p[i],j,Dimension);
00804       }
00805       
00806       /* Normalize the inequalities */
00807       for (i=NbEq;i<NbRows;i++) { 
00808         
00809         /* if (Mat->p[i][j] != 0) */
00810         if (value_notzero_p(p[i][j]))
00811           Combine(p[i],p[k],p[i],j,Dimension);
00812       }
00813     }
00814   } /* end of TRY */
00815   
00816   UNCATCH(any_exception_error);
00817   free(column_index), column_index = NULL;
00818   
00819   value_clear(gcd);
00820   return Rank;
00821 } /* Gauss */

00827  * The function returns the rank of the matrix 'Mat'.

00842   Matrix_Print(stderr,0,Mat);

00848 /*
00849  * Given 'Mat' - a matrix of equations and inequalities, 'Ray' - a matrix of 
00850  * lines and rays, 'Sat' - the corresponding saturation matrix, and 'Filter'
00851  * - an array to mark (with 1) the non-redundant equalities and inequalities, 
00852  * compute a polyhedron composed of 'Mat' as constraint matrix and 'Ray' as 
00853  * ray matrix after reductions. This function is usually called as a follow
00854  * up to 'Chernikova' to remove redundant constraints or rays.
00855  * Note: (1) 'Chernikova' ensures that there are no redundant lines and rays. 
00856  *       (2) The same function can be used with constraint and ray matrix used
00857   interchangbly.
00858  */ 
00859 static Polyhedron *Remove_Redundants(Matrix *Mat,Matrix *Ray,SatMatrix *Sat,unsigned *Filter) { 
00860   
00861   int i, j, k;
00862   unsigned Dimension, sat_nbcolumns, NbRay, NbConstraints, RowSize2,
00863            *Trace = NULL, *bx = NULL, *jx = NULL, Dim_RaySpace, b;
00864   unsigned NbBid, NbUni, NbEq, NbIneq;
00865   unsigned NbBid2, NbUni2, NbEq2, NbIneq2;
00866   int Redundant;
00867   int aux, *temp2 = NULL;
00868   Polyhedron *Pol = NULL;
00869   Vector *temp1 = NULL;
00870   unsigned Status;
00871   
00872   Dimension = Mat->NbColumns-1;     /* Homogeneous Dimension */
00873   NbRay = Ray->NbRows;
00874   sat_nbcolumns = Sat->NbColumns;
00875   NbConstraints = Mat->NbRows;
00876   RowSize2=sat_nbcolumns * sizeof(int);
00877   
00878   temp1 = Vector_Alloc(Dimension+1);
00879   if (!temp1) {
00880     errormsg1("Remove_Redundants", "outofmem", "out of memory space");
00881     return 0;
00882   }
00883 
00884   if (Filter) {
00885     temp2 = (int *)calloc(sat_nbcolumns, sizeof(int));
00886     if (!temp2)
00887       goto oom;
00888   }
00889   
00890   /* Introduce indirections into saturation matrix 'Sat' to simplify */
00891   /* processing with 'Sat' and allow easy exchanges of columns.      */
00892   bx = (unsigned *)malloc(NbConstraints * sizeof(unsigned));
00893   if (!bx)
00894     goto oom;
00895   jx = (unsigned *)malloc(NbConstraints * sizeof(unsigned));
00896   if (!jx)
00897     goto oom;
00898   CATCH(any_exception_error) {  
00899     
00900     Vector_Free(temp1);
00901     if (temp2) free(temp2);
00902     if (bx) free(bx);
00903     if (jx) free(jx);
00904     if (Trace) free(Trace);
00905     if (Pol) Polyhedron_Free(Pol);
00906 
00907     RETHROW();
00908   }
00909   TRY {
00910     
00911     /* For each constraint 'j' following mapping is defined to facilitate  */
00912     /* data access from saturation matrix 'Sat' :-                         */
00913     /* (1) jx[j] -> floor[j/(8*sizeof(int))]                               */
00914     /* (2) bx[j] -> bin(00..10..0) where position of 1 = j%(8*sizeof(int)) */
00915     
00916     i = 0;
00917     b = MSB;
00918     for (j=0; j<NbConstraints; j++) { 
00919       jx[j] = i;
00920       bx[j] = b;
00921       NEXT(i,b);
00922     }
00923     
00924     /* 
00925      * STEP(0): Count the number of vertices among the rays while initializing
00926      * the ray status count to 0. If no vertices are found, quit the procedure
00927      * and return an empty polyhedron as the result. 
00928      */            
00929     
00930     /* Reset the status element of each ray to zero. Store the number of  */
00931     /* vertices in 'aux'.                                                 */
00932     aux = 0;
00933     for (i=0; i<NbRay; i++) {  
00934       
00935       /* Ray->p[i][0] = 0 */
00936       value_set_si(Ray->p[i][0],0);
00937       
00938       /* If ray(i) is a vertex of the Inhomogenous system */
00939       if (value_notzero_p(Ray->p[i][Dimension]))
00940         aux++;              
00941     }
00942     
00943     /* If no vertices, return an empty polyhedron. */
00944     if (!aux)
00945       goto empty;
00946     
00947 #ifdef POLY_RR_DEBUG
00948     fprintf(stderr, "[Remove_redundants : Init]\nConstraints =");
00949     Matrix_Print(stderr,0,Mat);
00950     fprintf(stderr, "\nRays =");
00951     Matrix_Print(stderr,0,Ray);
00952 #endif
00953     
00954     /* 
00955      * STEP(1): Compute status counts for both rays and inequalities. For each
00956      * constraint, count the number of vertices/rays saturated by that 
00957      * constraint, and put the result in the status words. At the same time, 
00958      * for each vertex/ray, count the number of constraints saturated by it.
00959      * Delete any positivity constraints, but give rays credit in their status
00960      * counts for saturating the positivity constraint.
00961      */
00962     
00963     NbEq=0;
00964     
00965 #ifdef POLY_RR_DEBUG
00966     fprintf (stderr, " j = ");
00967 #endif
00968     
00969     for (j=0; j<NbConstraints; j++) {
00970       
00971 #ifdef POLY_RR_DEBUG
00972       fprintf (stderr, " %i ", j);
00973       fflush (stderr);
00974 #endif
00975       
00976       /* If constraint(j) is an equality, mark '1' in array 'temp2' */
00977       if (Filter && value_zero_p(Mat->p[j][0]))  
00978         temp2[jx[j]] |= bx[j]; 
00979       /* Reset the status element of each constraint to zero */
00980       value_set_si(Mat->p[j][0],0);
00981       
00982       /* Identify and remove the positivity constraint 1>=0 */
00983       i = First_Non_Zero(Mat->p[j]+1, Dimension-1);
00984       
00985 #ifdef POLY_RR_DEBUG
00986       fprintf(stderr, "[Remove_redundants : IntoStep1]\nConstraints =");
00987       Matrix_Print(stderr,0,Mat);
00988       fprintf (stderr, " j = %i \n", j);
00989 #endif
00990       
00991       /* Check if constraint(j) is a positivity constraint, 1 >= 0, or if it */
00992       /* is 1==0. If constraint(j) saturates all the rays of the matrix 'Ray'*/
00993       /* then it is an equality. in this case, return an empty polyhedron.   */
00994       
00995       if (i == -1) {
00996         for (i=0; i<NbRay; i++)
00997           if (!(Sat->p[i][jx[j]]&bx[j])) {
00998             
00999             /* Mat->p[j][0]++ */
01000             value_increment(Mat->p[j][0],Mat->p[j][0]);
01001           }
01002         
01003         /* if ((Mat->p[j][0] == NbRay) &&   : it is an equality
01004            (Mat->p[j][Dimension] != 0)) : and its not 0=0 */
01005         if ((value_cmp_si(Mat->p[j][0], NbRay) == 0) &&
01006             (value_notzero_p(Mat->p[j][Dimension])))
01007           goto empty;
01008         
01009         /* Delete the positivity constraint */
01010         NbConstraints--;
01011         if (j==NbConstraints) continue;
01012         Vector_Exchange(Mat->p[j], Mat->p[NbConstraints], temp1->Size);
01013         exchange(jx[j], jx[NbConstraints], aux);
01014         exchange(bx[j], bx[NbConstraints], aux);
01015         j--; continue;
01016       }
01017       
01018       /* Count the number of vertices/rays saturated by each constraint. At  */
01019       /* the same time, count the number of constraints saturated by each ray*/
01020       for (i=0; i<NbRay; i++) 
01021         if (!(Sat->p[i][jx[j]]&bx[j])) {  
01022           
01023           /* Mat->p[j][0]++ */
01024           value_increment(Mat->p[j][0],Mat->p[j][0]);
01025           
01026           /* Ray->p[i][0]++ */
01027           value_increment (Ray->p[i][0],Ray->p[i][0]);
01028         }
01029       
01030       /* if (Mat->p[j][0]==NbRay) then increment the number of eq. count */
01031       if (value_cmp_si(Mat->p[j][0], NbRay) == 0)
01032         NbEq++;    /* all vertices/rays are saturated */
01033     }
01034     Mat->NbRows = NbConstraints;
01035     
01036     NbBid=0;
01037     for (i=0; i<NbRay; i++) {
01038       
01039       /* Give rays credit for saturating the positivity constraint */
01040       if (value_zero_p(Ray->p[i][Dimension]))   
01041         
01042         /* Ray->p[i][0]++ */
01043         value_increment(Ray->p[i][0],Ray->p[i][0]);
01044       
01045       /* If ray(i) saturates all the constraints including positivity  */
01046       /* constraint then it is a bi-directional ray or line. Increment */
01047       /* 'NbBid' by one.                                               */
01048       
01049       /* if (Ray->p[i][0]==NbConstraints+1) */
01050       if (value_cmp_si(Ray->p[i][0], NbConstraints+1) == 0)
01051         NbBid++;
01052     }
01053     
01054 #ifdef POLY_RR_DEBUG
01055     fprintf(stderr, "[Remove_redundants : Step1]\nConstraints =");
01056     Matrix_Print(stderr,0,Mat);
01057     fprintf(stderr, "\nRay =");
01058     Matrix_Print(stderr,0,Ray);
01059 #endif
01060     
01061     /* 
01062      * STEP(2): Sort equalities to the top of constraint matrix 'Mat'. Detect
01063      * implicit equations such as y>=3; y<=3. Keep Inequalities in same 
01064      * relative order. (Note: Equalities are constraints which saturate all of
01065      * the rays)              
01066      */
01067     
01068     for (i=0; i<NbEq; i++) {
01069       
01070       /* If constraint(i) doesn't saturate some ray, then it is an inequality*/
01071       if (value_cmp_si(Mat->p[i][0], NbRay) != 0) {
01072         
01073         /* Skip over inequalities and find an equality */
01074         for (k=i+1; value_cmp_si(Mat->p[k][0], NbRay) != 0 && k < NbConstraints; k++)
01075           ;
01076         if (k==NbConstraints) /* If none found then error */ break;
01077         
01078         /* Slide inequalities down the array 'Mat' and move equality up to */
01079         /* position 'i'.                                                   */
01080         Vector_Copy(Mat->p[k], temp1->p, temp1->Size);
01081         aux = jx[k];
01082         j   = bx[k];
01083         for (;k>i;k--) {  
01084           Vector_Copy(Mat->p[k-1], Mat->p[k], temp1->Size);
01085           jx[k] = jx[k-1];
01086           bx[k] = bx[k-1];
01087         }
01088         Vector_Copy(temp1->p, Mat->p[i], temp1->Size);
01089         jx[i] = aux;
01090         bx[i] = j;
01091       }
01092     }
01093 
01094     /* for SIMPLIFY */
01095     if (Filter) {
01096       Value mone;
01097       value_init(mone);
01098       value_set_si(mone, -1);
01099       /* Normalize equalities to have lexpositive coefficients to
01100        * be able to detect identical equalities.
01101        */
01102       for (i = 0; i < NbEq; i++) {
01103         int pos = First_Non_Zero(Mat->p[i]+1, Dimension);
01104         if (pos == -1)
01105           continue;
01106         if (value_neg_p(Mat->p[i][1+pos]))
01107           Vector_Scale(Mat->p[i]+1, Mat->p[i]+1, mone, Dimension);
01108       }
01109       value_clear(mone);
01110       for (i=0; i<NbEq; i++) {
01111         
01112         /* Detect implicit constraints such as y>=3 and y<=3 */
01113         Redundant = 0;
01114         for (j=i+1; j<NbEq; j++) {
01115           /* Only check equalities, i.e., 'temp2' has entry 1             */
01116           if (!(temp2[jx[j]] & bx[j]))
01117             continue;
01118           /* Redundant if both are same `and' constraint(j) was equality. */
01119           if (Vector_Equal(Mat->p[i]+1, Mat->p[j]+1, Dimension)) {
01120             Redundant=1; 
01121             break;
01122           }
01123         }
01124         
01125         /* Set 'Filter' entry to 1 corresponding to the irredundant equality*/
01126         if (!Redundant) Filter[jx[i]] |= bx[i];  /* set flag */
01127       }
01128     }
01129 
01130 #ifdef POLY_RR_DEBUG
01131     fprintf(stderr, "[Remove_redundants : Step2]\nConstraints =");
01132     Matrix_Print(stderr,0,Mat);
01133     fprintf(stderr, "\nRay =");
01134     Matrix_Print(stderr,0,Ray);
01135 #endif
01136     
01137     /* 
01138      * STEP(3): Perform Gaussian elimiation on the list of equalities. Obtain
01139      * a minimal basis by solving for as many variables as possible. Use this 
01140      * solution to reduce the inequalities by eliminating as many variables as
01141      * possible. Set NbEq2 to the rank of the system of equalities.
01142      */
01143     
01144     NbEq2 = Gauss(Mat,NbEq,Dimension);
01145     
01146     /* If number of equalities is not less then the homogenous dimension, */
01147     /* return an empty polyhedron.                                        */
01148     
01149     if (NbEq2 >= Dimension)
01150       goto empty;
01151     
01152 #ifdef POLY_RR_DEBUG
01153     fprintf(stderr, "[Remove_redundants : Step3]\nConstraints =");
01154     Matrix_Print(stderr,0,Mat);
01155     fprintf(stderr, "\nRay =");
01156     Matrix_Print(stderr,0,Ray);
01157 #endif
01158     
01159     /*
01160      * STEP(4): Sort lines to the top of ray matrix 'Ray', leaving rays
01161      * afterwards. Detect implicit lines such as ray(1,2) and ray(-1,-2). 
01162      * (Note: Lines are rays which saturate all of the constraints including
01163      * the positivity constraint 1>=0. 
01164      */
01165     
01166     
01167     for (i=0, k=NbRay; i<NbBid && k>i; i++) {
01168       /* If ray(i) doesn't saturate some constraint then it is not a line */
01169       if (value_cmp_si(Ray->p[i][0], NbConstraints+1) != 0) {
01170         
01171         /* Skip over rays and vertices and find a line (bi-directional rays) */
01172         while (--k > i && value_cmp_si(Ray->p[k][0], NbConstraints+1) != 0)
01173           ;
01174         
01175         /* Exchange positions of ray(i) and line(k), thus sorting lines to */
01176         /* the top of matrix 'Ray'.                                        */
01177         Vector_Exchange(Ray->p[i], Ray->p[k], temp1->Size);
01178         bexchange(Sat->p[i], Sat->p[k], RowSize2);
01179       }     
01180     }   
01181 
01182 #ifdef POLY_RR_DEBUG
01183     fprintf(stderr, "[Remove_redundants : Step4]\nConstraints =");
01184     Matrix_Print(stderr,0,Mat);
01185     fprintf(stderr, "\nRay =");
01186     Matrix_Print(stderr,0,Ray);
01187 #endif
01188     
01189     /* 
01190      * STEP(5): Perform Gaussian elimination on the lineality space to obtain
01191      * a minimal basis of lines. Use this basis to reduce the representation
01192      * of the uniderectional rays. Set 'NbBid2' to the rank of the system of 
01193      * lines. 
01194      */
01195     
01196     NbBid2 = Gauss(Ray, NbBid, Dimension);
01197     
01198 #ifdef POLY_RR_DEBUG
01199     fprintf(stderr, "[Remove_redundants : After Gauss]\nRay =");
01200     Matrix_Print(stderr,0,Ray);
01201 #endif
01202     
01203     /* If number of lines is not less then the homogenous dimension, return */
01204     /* an empty polyhedron.                                                 */
01205     if (NbBid2>=Dimension) {
01206       errormsg1("RemoveRedundants", "rmrdt", "dimension error");
01207       goto empty;
01208     }
01209     
01210     /* Compute dimension of non-homogenous ray space */
01211     Dim_RaySpace = Dimension-1-NbEq2-NbBid2;  
01212     
01213 #ifdef POLY_RR_DEBUG
01214     fprintf(stderr, "[Remove_redundants : Step5]\nConstraints =");
01215     Matrix_Print(stderr,0,Mat);
01216     fprintf(stderr, "\nRay =");
01217     Matrix_Print(stderr,0,Ray);
01218 #endif
01219     
01220     /* 
01221      * STEP(6): Do a first pass filter of inequalities and equality identifi-
01222      * cation. New positivity constraints may have been created by step(3). 
01223      * Check for and eliminate them. Count the irredundant inequalities and 
01224      * store count in 'NbIneq'.  
01225      */
01226  
01227     NbIneq=0;
01228     for (j=0; j<NbConstraints; j++) {
01229       
01230       /* Identify and remove the positivity constraint 1>=0 */
01231       i = First_Non_Zero(Mat->p[j]+1, Dimension-1);
01232       
01233       /* Check if constraint(j) is a positivity constraint, 1>= 0, or if it */
01234       /* is 1==0.                                                           */
01235       if (i == -1) {
01236         /* if ((Mat->p[j][0]==NbRay) &&   : it is an equality 
01237            (Mat->p[j][Dimension]!=0))    : and its not 0=0 */
01238         if ((value_cmp_si(Mat->p[j][0], NbRay) == 0) &&
01239             (value_notzero_p(Mat->p[j][Dimension])))
01240           goto empty;
01241         
01242         /* Set the positivity constraint redundant by setting status element */
01243         /* equal to 2.                                                       */
01244         value_set_si(Mat->p[j][0],2);
01245         continue;
01246       }
01247       
01248       Status = VALUE_TO_INT(Mat->p[j][0]);
01249       
01250       if (Status == 0)  
01251         value_set_si(Mat->p[j][0], 2);  /* constraint is redundant */
01252       else if (Status < Dim_RaySpace)   
01253         value_set_si(Mat->p[j][0], 2);  /* constraint is redundant */
01254       else if (Status == NbRay)         
01255         value_set_si(Mat->p[j][0], 0);  /* constraint is an equality */
01256       else { 
01257         NbIneq++;                       /* constraint is an irredundant inequality */ 
01258         value_set_si(Mat->p[j][0], 1);  /* inequality */
01259       }
01260     }
01261     
01262 #ifdef POLY_RR_DEBUG
01263     fprintf(stderr, "[Remove_redundants : Step6]\nConstraints =");
01264     Matrix_Print(stderr,0,Mat);
01265     fprintf(stderr, "\nRay =");
01266     Matrix_Print(stderr,0,Ray);
01267 #endif
01268     
01269     /* 
01270      * STEP(7): Do a first pass filter of rays and identification of lines.
01271      * Count the irredundant Rays and store count in 'NbUni'. 
01272      */
01273     
01274     NbUni=0;
01275     for (j=0; j<NbRay; j++) { 
01276       Status = VALUE_TO_INT(Ray->p[j][0]);
01277       
01278       if (Status < Dim_RaySpace)        
01279         value_set_si(Ray->p[j][0], 2);  /* ray is redundant */
01280       else if (Status == NbConstraints+1) 
01281         value_set_si(Ray->p[j][0], 0);  /* ray is a line */
01282       else {
01283         NbUni++;                        /* an irredundant unidirectional ray. */
01284         value_set_si(Ray->p[j][0], 1);  /* ray */
01285       }
01286     }
01287     
01288     /*
01289      * STEP(8): Create the polyhedron (using approximate sizes).
01290      * Number of constraints = NbIneq + NbEq2 + 1
01291      * Number of rays = NbUni + NbBid2 
01292      * Partially fill the polyhedron structure with the lines computed in step
01293      * 3 and the equalities computed in step 5. 
01294      */
01295     
01296     Pol = Polyhedron_Alloc(Dimension-1, NbIneq+NbEq2+1, NbUni+NbBid2);
01297     if (!Pol) {
01298       UNCATCH(any_exception_error);
01299       goto oom;
01300     }
01301     Pol->NbBid = NbBid2;
01302     Pol->NbEq  = NbEq2;
01303     
01304     /* Partially fill the polyhedron structure */
01305     if (NbBid2) Vector_Copy(Ray->p[0], Pol->Ray[0], (Dimension+1)*NbBid2);
01306     if (NbEq2)  Vector_Copy(Mat->p[0], Pol->Constraint[0], (Dimension+1)*NbEq2);
01307     
01308 #ifdef POLY_RR_DEBUG
01309     fprintf(stderr, "[Remove_redundants : Step7]\nConstraints =");
01310     Matrix_Print(stderr,0,Mat);
01311     fprintf(stderr, "\nRay =");
01312     Matrix_Print(stderr,0,Ray);
01313 #endif
01314     
01315     /* 
01316      * STEP(9): Final Pass filter of inequalities and detection of redundant
01317      * inequalities. Redundant inequalities include: 
01318      * (1) Inequalities which are always true, such as 1>=0, 
01319      * (2) Redundant inequalities such as y>=4 given y>=3, or x>=1 given x=2. 
01320      * (3) Redundant inequalities such as x+y>=5 given x>=3 and y>=2.
01321      * Every 'good' inequality must saturate at least 'Dimension' rays and be 
01322      * unique.
01323      */
01324     
01325     /* 'Trace' is a (1 X sat_nbcolumns) row matrix to hold the union of all */
01326     /* rows (corresponding to irredundant rays) of saturation matrix 'Sat'  */
01327     /* which saturate some constraint 'j'. See figure below:-               */
01328     Trace=(unsigned *)malloc(sat_nbcolumns * sizeof(unsigned));
01329     if(!Trace) {
01330       UNCATCH(any_exception_error);
01331       goto oom;
01332     }
01333     
01334     /*                         NbEq      NbConstraints
01335                                |----------->
01336                                 ___________j____
01337                                |           |   |
01338                                |      Mat  |   |
01339                                |___________|___|
01340                                            |                  
01341      NbRay  ^ ________         ____________|____
01342             | |-------|--------|-----------0---|t1
01343             |i|-------|--------|-----------0---|t2
01344             | | Ray   |        |    Sat        |
01345      NbBid  - |-------|--------|-----------0---|tk
01346               |_______|        |_______________|
01347                                            |
01348                                            |
01349                                       -OR- (of rows t1,t2,...,tk)
01350                                ________|___|____
01351                                |_____Trace_0___|
01352                                
01353     */
01354     
01355     NbIneq2 = 0;
01356     for (j=NbEq; j<NbConstraints; j++) {
01357       
01358       /* if (Mat->p[j][0]==1) : non-redundant inequality */
01359       if (value_one_p (Mat->p[j][0])) { 
01360         for (k=0; k<sat_nbcolumns; k++) Trace[k]=0;  /* init Trace */
01361         
01362         /* Compute Trace: the union of all rows of Sat where constraint(j) */
01363         /* is saturated.                                                   */
01364         for (i=NbBid; i<NbRay; i++) 
01365           
01366           /* if (Ray->p[i][0]==1) */
01367           if (value_one_p(Ray->p[i][0])) { 
01368             if (!(Sat->p[i][jx[j]]&bx[j])) 
01369               for (k=0; k<sat_nbcolumns; k++) Trace[k] |= Sat->p[i][k];
01370           }
01371         
01372         /* Only constraint(j) should saturate this set of vertices/rays. */
01373         /* If another non-redundant constraint also saturates this set,  */
01374         /* then constraint(j) is redundant                               */
01375         Redundant=0;
01376         for (i=NbEq; i<NbConstraints; i++) {
01377           
01378           /* if ((Mat->p[i][0] ==1) && (i!=j) && !(Trace[jx[i]] & bx[i]) ) */
01379           if (value_one_p(Mat->p[i][0]) && (i!=j) && !(Trace[jx[i]] & bx[i])) {
01380             Redundant=1;
01381             break;
01382           }
01383         }
01384         if (Redundant) {
01385           value_set_si(Mat->p[j][0],2);
01386         }       
01387         else {
01388           Vector_Copy(Mat->p[j], Pol->Constraint[NbEq2+NbIneq2], Dimension+1);
01389           if (Filter) Filter[jx[j]] |= bx[j];           /* for SIMPLIFY */
01390           NbIneq2++;
01391         }
01392       }
01393     }
01394     free(Trace), Trace = NULL;
01395     
01396 #ifdef POLY_RR_DEBUG
01397     fprintf(stderr, "[Remove_redundants : Step8]\nConstraints =");
01398     Matrix_Print(stderr,0,Mat);
01399     fprintf(stderr, "\nRay =");
01400     Matrix_Print(stderr,0,Ray);
01401 #endif
01402     
01403     /* 
01404      * Step(10): Final pass filter of rays and detection of redundant rays.
01405      * The final list of rays is written to polyhedron.                     
01406      */
01407     
01408     /* Trace is a (NbRay x 1) column matrix to hold the union of all columns */
01409     /* (corresponding to irredundant inequalities) of saturation matrix 'Sat'*/
01410     /* which saturate some ray 'i'. See figure below:-                       */
01411     
01412     Trace=(unsigned *)malloc(NbRay * sizeof(unsigned));
01413     if(!Trace) {
01414       UNCATCH(any_exception_error);
01415       goto oom;
01416     }
01417     
01418     /*                     NbEq     NbConstraints
01419                              |---------->
01420                         ___________j_____
01421                         |      | |   |  |
01422                         |      Mat   |  |
01423                         |______|_|___|__|
01424                                | |   |
01425 NbRay ^ _________       _______|_|___|__   ___
01426       | |       |       |      | |   |  |  |T|
01427       | |  Ray  |       |   Sat| |   |  |  |r|
01428       | |       |       |      | |   |  |  |a|  Trace = Union[col(t1,t2,..,tk)]
01429       |i|-------|------>i      0 0   0  |  |c|
01430 NbBid - |       |       |      | |   |  |  |e|
01431         |_______|       |______|_|___|__|  |_|
01432                               t1 t2  tk
01433     */    
01434   
01435     NbUni2 = 0;
01436     
01437     /* Let 'aux' be the number of rays not vertices */ 
01438     aux = 0;     
01439     for (i=NbBid; i<NbRay; i++) {
01440       
01441       /* if (Ray->p[i][0]==1) */
01442       if (value_one_p (Ray->p[i][0])) { 
01443         
01444         /* if (Ray->p[i][Dimension]!=0) : vertex */
01445         if (value_notzero_p (Ray->p[i][Dimension]))
01446           for (k=NbBid; k<NbRay; k++) Trace[k]=0;         /* init Trace */
01447         else /* for ray */
01448           
01449           /* Include the positivity constraint incidences for rays. The */
01450           /* positivity constraint saturates all rays and no vertices   */
01451           
01452           for (k=NbBid; k<NbRay; k++)
01453             
01454             /* Trace[k]=(Ray->p[k][Dimension]!=0); */
01455             Trace[k] = (value_notzero_p (Ray->p[k][Dimension]));
01456         
01457         /* Compute Trace: the union of all columns of Sat where ray(i) is  */
01458         /* saturated.                                                      */
01459         for (j=NbEq; j<NbConstraints; j++)
01460           
01461           /* if (Mat->p[j][0]==1) : inequality */
01462           if (value_one_p (Mat->p[j][0])) { 
01463             if (!(Sat->p[i][jx[j]]&bx[j]))
01464               for (k=NbBid; k<NbRay; k++) Trace[k] |= Sat->p[k][jx[j]]&bx[j];
01465           }
01466         
01467         /* If ray i does not saturate any inequalities (other than the   */
01468         /* the positivity constraint, then it is the case that there is  */
01469         /* only one inequality and that ray is its orthogonal            */
01470         
01471         /* only ray(i) should saturate this set of inequalities. If      */
01472         /* another non-redundant ray also saturates this set, then ray(i)*/
01473         /* is redundant                                                  */
01474         
01475         Redundant = 0;
01476         for (j=NbBid; j<NbRay; j++) { 
01477           
01478           /* if ( (Ray->p[j][0]==1) && (i!=j) && !Trace[j] ) */
01479           if (value_one_p (Ray->p[j][0]) && (i!=j) && !Trace[j]) { 
01480             Redundant=1;
01481             break;
01482           }
01483         }
01484         if (Redundant) 
01485           value_set_si(Ray->p[i][0],2); 
01486         else {
01487           Vector_Copy(Ray->p[i], Pol->Ray[NbBid2+NbUni2], Dimension+1);
01488           NbUni2++;  /* Increment number of uni-directional rays */
01489           
01490           /* if (Ray->p[i][Dimension]==0) */ 
01491           if (value_zero_p (Ray->p[i][Dimension]))
01492             aux++; /* Increment number of rays which are not vertices */
01493         }
01494       }
01495     }
01496     
01497     /* Include the positivity constraint */
01498     if (aux>=Dim_RaySpace) {
01499       Vector_Set(Pol->Constraint[NbEq2+NbIneq2],0,Dimension+1);
01500       value_set_si(Pol->Constraint[NbEq2+NbIneq2][0],1);
01501       value_set_si(Pol->Constraint[NbEq2+NbIneq2][Dimension],1);
01502       NbIneq2++;
01503     }   
01504   } /* end of TRY */
01505   
01506   UNCATCH(any_exception_error);
01507   
01508 #ifdef POLY_RR_DEBUG
01509   fprintf(stderr, "[Remove_redundants : Step9]\nConstraints =");
01510   Matrix_Print(stderr,0,Mat);
01511   fprintf(stderr, "\nRay =");
01512   Matrix_Print(stderr,0,Ray);
01513 #endif
01514   
01515   free(Trace);
01516   free(bx);
01517   free(jx);
01518   if (temp2)
01519     free(temp2);
01520   
01521   Pol->NbConstraints = NbEq2 + NbIneq2;
01522   Pol->NbRays = NbBid2 + NbUni2;
01523   
01524   Vector_Free(temp1);
01525   F_SET(Pol, 
01526         POL_VALID | POL_INEQUALITIES | POL_FACETS | POL_POINTS | POL_VERTICES);
01527   return Pol;
01528 
01529 oom:
01530   errormsg1("Remove_Redundants", "outofmem", "out of memory space");
01531 
01532   Vector_Free(temp1);
01533   if (temp2)
01534     free(temp2);
01535   if (bx)
01536     free(bx);
01537   if (jx)
01538     free(jx);
01539   return NULL;
01540 
01541 empty:
01542   Vector_Free(temp1);
01543   if (temp2)
01544     free(temp2);
01545   if (bx)
01546     free(bx);
01547   if (jx)
01548     free(jx);
01549   UNCATCH(any_exception_error);
01550   return Empty_Polyhedron(Dimension-1);
01551 } /* Remove_Redundants */

01738 /* 
01739  * Create and return a universe polyhedron of non-homogenous dimension
01740  * 'Dimension'. A universe polyhedron is characterized by :-
01741  * (a) The dimension of rayspace is zero. 
01742  * (b) The dimension of lineality space is the dimension of the polyhedron.
01743  * (c) There is only one constraint (positivity constraint) in the constraint
01744  *     set given by : 1 >= 0. 
01745  * (d) The bi-directional ray set is the canonical set of vectors. 
01746  * (e) The only vertex is the origin (0,0,0,....0).  
01747  */

01781 /*
01782 
01783 Sort constraints and remove trivially redundant constraints.
01784 
01785 */
01786 static void SortConstraints(Matrix *Constraints, unsigned NbEq)
01787 {
01788     int i, j, k;
01789     for (i = NbEq; i < Constraints->NbRows; ++i) {
01790         int max = i;
01791         for (k = i+1; k < Constraints->NbRows; ++k) {
01792             for (j = 1; j < Constraints->NbColumns-1; ++j) {
01793                 if (value_eq(Constraints->p[k][j],
01794                              Constraints->p[max][j]))
01795                     continue;
01796                 if (value_abs_lt(Constraints->p[k][j],
01797                                  Constraints->p[max][j]))
01798                     break;
01799                 if (value_abs_eq(Constraints->p[k][j],
01800                                  Constraints->p[max][j]) &&
01801                     value_pos_p(Constraints->p[max][j]))
01802                         break;
01803                 max = k;
01804                 break;
01805             }
01806             /* equal, except for possibly the constant
01807              * => remove constraint with biggest constant
01808              */
01809             if (j == Constraints->NbColumns-1) {
01810                 if (value_lt(Constraints->p[k][j], Constraints->p[max][j]))
01811                     Vector_Exchange(Constraints->p[k], 
01812                                     Constraints->p[max], 
01813                                     Constraints->NbColumns);
01814                 Constraints->NbRows--;
01815                 if (k < Constraints->NbRows)
01816                     Vector_Exchange(Constraints->p[k], 
01817                                     Constraints->p[Constraints->NbRows], 
01818                                     Constraints->NbColumns);
01819                 k--;
01820             }
01821         }
01822         if (max != i)
01823             Vector_Exchange(Constraints->p[max], Constraints->p[i], 
01824                             Constraints->NbColumns);
01825     }
01826 }

01828 /*
01829 
01830 Search for trivial implicit equalities,
01831 assuming the constraints have been sorted.
01832 
01833 */
01834 
01835 static int ImplicitEqualities(Matrix *Constraints, unsigned NbEq)
01836 {
01837     int row, nrow, k;
01838     int found = 0;
01839     Value tmp;
01840     for (row = NbEq; row < Constraints->NbRows; ++row) {
01841         int d = First_Non_Zero(Constraints->p[row]+1, Constraints->NbColumns-2);
01842         if (d == -1) {
01843             /* -n >= 0 */
01844             if (value_neg_p(Constraints->p[row][Constraints->NbColumns-1])) {
01845                 value_set_si(Constraints->p[row][0], 0);
01846                 found = 1;
01847             }
01848             break;
01849         }
01850         if (value_neg_p(Constraints->p[row][1+d]))
01851             continue;
01852         for (nrow = row+1; nrow < Constraints->NbRows; ++nrow) {
01853             if (value_zero_p(Constraints->p[nrow][1+d]))
01854                 break;
01855             if (value_pos_p(Constraints->p[nrow][1+d]))
01856                 continue;
01857             for (k = d; k < Constraints->NbColumns-1; ++k) {
01858                 if (value_abs_ne(Constraints->p[row][1+k], 
01859                                  Constraints->p[nrow][1+k]))
01860                     break;
01861                 if (value_zero_p(Constraints->p[row][1+k]))
01862                     continue;
01863                 if (value_eq(Constraints->p[row][1+k], Constraints->p[nrow][1+k]))
01864                     break;
01865             }
01866             if (k == Constraints->NbColumns-1) {
01867                 value_set_si(Constraints->p[row][0], 0);
01868                 value_set_si(Constraints->p[nrow][0], 0);
01869                 found = 1;
01870                 break;
01871             }
01872             if (k != Constraints->NbColumns-2)
01873                 continue;
01874             /* if the constants are such that 
01875              * the sum c1+c2 is negative then the constraints conflict
01876              */
01877             value_init(tmp);
01878             value_addto(tmp, Constraints->p[row][1+k], 
01879                              Constraints->p[nrow][1+k]);
01880             if (value_sign(tmp) < 0) {
01881                 Vector_Set(Constraints->p[row], 0, Constraints->NbColumns-1);
01882                 Vector_Set(Constraints->p[nrow], 0, Constraints->NbColumns-1);
01883                 value_set_si(Constraints->p[row][1+k], 1);
01884                 value_set_si(Constraints->p[nrow][1+k], 1);
01885                 found = 1;
01886             }
01887             value_clear(tmp);
01888             if (found)
01889                 break;
01890         }
01891     }
01892     return found;
01893 }

01895 /**
01896 
01897 Given a matrix of constraints ('Constraints'), construct and return a 
01898 polyhedron.
01899 
01900 @param Constraints Constraints (may be modified!)
01901 @param NbMaxRays Estimated number of rays in the ray matrix of the
01902 polyhedron.
01903 @return newly allocated Polyhedron
01904 
01905 */ 
01906 Polyhedron *Constraints2Polyhedron(Matrix *Constraints,unsigned NbMaxRays) {
01907   
01908   Polyhedron *Pol = NULL;
01909   Matrix *Ray = NULL;
01910   SatMatrix *Sat = NULL;
01911   unsigned Dimension, nbcolumns;
01912   int i;
01913 
01914   Dimension = Constraints->NbColumns - 1;  /* Homogeneous Dimension */
01915   if (Dimension < 1) {
01916     errormsg1("Constraints2Polyhedron","invalidpoly","invalid polyhedron dimension");
01917     return 0;
01918   }
01919   
01920   /* If there is no constraint in the constraint matrix, return universe */
01921   /* polyhderon.                                                         */
01922   if (Constraints->NbRows==0) {  
01923     Pol = Universe_Polyhedron(Dimension-1);
01924     return Pol;
01925   }
01926 
01927   if (POL_ISSET(NbMaxRays, POL_NO_DUAL)) {
01928     unsigned NbEq;
01929     unsigned Rank;
01930     Value tmp;
01931     if (POL_ISSET(NbMaxRays, POL_INTEGER))
01932       value_init(tmp);
01933     do {
01934       NbEq = 0;
01935       /* Move equalities up */
01936       for (i = 0; i < Constraints->NbRows; ++i)
01937         if (value_zero_p(Constraints->p[i][0])) {
01938           if (POL_ISSET(NbMaxRays, POL_INTEGER) &&
01939             ConstraintSimplify(Constraints->p[i], 
01940                                Constraints->p[i], Dimension+1, &tmp)) {
01941             value_clear(tmp);
01942             return Empty_Polyhedron(Dimension-1);
01943           }
01944           /* detect 1 == 0, possibly created by ImplicitEqualities */
01945           if (First_Non_Zero(Constraints->p[i]+1, Dimension-1) == -1 &&
01946               value_notzero_p(Constraints->p[i][Dimension])) {
01947             if (POL_ISSET(NbMaxRays, POL_INTEGER))
01948               value_clear(tmp);
01949             return Empty_Polyhedron(Dimension-1);
01950           }
01951           if (i != NbEq)
01952             ExchangeRows(Constraints, i, NbEq);
01953           ++NbEq;
01954         }
01955       Rank = Gauss(Constraints, NbEq, Dimension);
01956       if (POL_ISSET(NbMaxRays, POL_INTEGER))
01957         for (i = NbEq; i < Constraints->NbRows; ++i)
01958           ConstraintSimplify(Constraints->p[i], 
01959                              Constraints->p[i], Dimension+1, &tmp);
01960       SortConstraints(Constraints, NbEq);
01961     } while (ImplicitEqualities(Constraints, NbEq));
01962     if (POL_ISSET(NbMaxRays, POL_INTEGER))
01963       value_clear(tmp);
01964     Pol = Polyhedron_Alloc(Dimension-1, Constraints->NbRows - (NbEq-Rank), 0);
01965     if (Rank > 0)
01966         Vector_Copy(Constraints->p[0], Pol->Constraint[0], 
01967                     Rank * Constraints->NbColumns);
01968     if (Constraints->NbRows > NbEq)
01969         Vector_Copy(Constraints->p[NbEq], Pol->Constraint[Rank], 
01970                     (Constraints->NbRows - NbEq) * Constraints->NbColumns);
01971     Pol->NbEq = Rank;
01972     /* Make sure nobody accesses the rays by accident */
01973     Pol->Ray = 0;
01974     F_SET(Pol, POL_VALID | POL_INEQUALITIES);
01975     return Pol;
01976   }
01977 
01978   if (Dimension > NbMaxRays)
01979     NbMaxRays = Dimension;
01980     
01981   /*
01982    * Rather than adding a 'positivity constraint', it is better to
01983    * initialize the lineality space with line in each of the index
01984    * dimensions, but no line in the lambda dimension. Then initialize
01985    * the ray space with an origin at 0.  This is what you get anyway,
01986    * after the positivity constraint has been processed by Chernikova
01987    * function.
01988    */
01989 
01990   /* Allocate and initialize the Ray Space */
01991   Ray = Matrix_Alloc(NbMaxRays, Dimension+1);
01992   if(!Ray) {
01993     errormsg1("Constraints2Polyhedron","outofmem","out of memory space");
01994     return 0;
01995   }
01996   Vector_Set(Ray->p_Init,0, NbMaxRays * (Dimension+1));
01997   for (i=0; i<Dimension; i++) {    
01998     
01999     /* Ray->p[i][i+1] = 1 */
02000     value_set_si(Ray->p[i][i+1],1);
02001   }
02002 
02003   /* Ray->p[Dimension-1][0] = 1 : mark for ray */
02004   value_set_si(Ray->p[Dimension-1][0],1);
02005   Ray->NbRows = Dimension; 
02006 
02007   /* Initialize the Sat Matrix */
02008   nbcolumns = (Constraints->NbRows - 1)/(sizeof(int)*8) + 1;
02009   Sat = SMAlloc(NbMaxRays, nbcolumns);
02010   SMVector_Init(Sat->p_init,Dimension*nbcolumns);
02011   Sat->NbRows = Dimension;
02012 
02013   CATCH(any_exception_error) {
02014 
02015     /* In case of overflow, free the allocated memory and forward. */    
02016     if (Sat) SMFree(&Sat);
02017     if (Ray) Matrix_Free(Ray);
02018     if (Pol) Polyhedron_Free(Pol);
02019     RETHROW();
02020   }
02021   TRY {
02022 
02023     /* Create ray matrix 'Ray' from constraint matrix 'Constraints' */
02024     Chernikova(Constraints,Ray,Sat,Dimension-1,NbMaxRays,0,0);
02025 
02026 #ifdef POLY_DEBUG
02027     fprintf(stderr, "[constraints2polyhedron]\nConstraints = ");
02028     Matrix_Print(stderr,0,Constraints);
02029     fprintf(stderr, "\nRay = ");
02030     Matrix_Print(stderr,0,Ray);
02031     fprintf(stderr, "\nSat = ");
02032     SMPrint(Sat);
02033 #endif
02034     
02035     /* Remove the redundant constraints and create the polyhedron */
02036     Pol = Remove_Redundants(Constraints,Ray,Sat,0);
02037   } /* end of TRY */
02038 
02039   UNCATCH(any_exception_error);
02040   
02041 #ifdef POLY_DEBUG
02042   fprintf(stderr, "\nPol = ");
02043   Polyhedron_Print(stderr,"%4d",Pol);
02044 #endif
02045   
02046   SMFree(&Sat), Sat = NULL;
02047   Matrix_Free(Ray), Ray = NULL; 
02048   return Pol;
02049 } /* Constraints2Polyhedron */

02074 /** 
02075 
02076 Given a matrix of rays 'Ray', create and return a polyhedron. 
02077 
02078 @param Ray Rays (may be modified!)
02079 @param NbMaxConstrs Estimated number of constraints in the polyhedron.
02080 @return newly allocated Polyhedron
02081 
02082 */ 
02083 Polyhedron *Rays2Polyhedron(Matrix *Ray,unsigned NbMaxConstrs) {
02084   
02085   Polyhedron *Pol = NULL;

02087   SatMatrix *Sat = NULL, *SatTranspose = NULL;
02088   unsigned Dimension, nbcolumns;
02089   int i;
02090   
02091   Dimension = Ray->NbColumns-1;        /* Homogeneous Dimension */
02092   Sat = NULL;
02093   SatTranspose = NULL;
02094   Mat = NULL;
02095   
02096   if (Ray->NbRows==0) {  
02097     
02098     /* If there is no ray in the matrix 'Ray', return an empty polyhedron */
02099     Pol = Empty_Polyhedron(Dimension-1);
02100     return(Pol);
02101   }
02102 
02103   /* Ignore for now */
02104   if (POL_ISSET(NbMaxConstrs, POL_NO_DUAL))
02105     NbMaxConstrs = 0;
02106 
02107   if (Dimension > NbMaxConstrs)
02108     NbMaxConstrs = Dimension;
02109   
02110   /* Allocate space for constraint matrix 'Mat' */
02111   Mat = Matrix_Alloc(NbMaxConstrs,Dimension+1);
02112   if(!Mat) {
02113     errormsg1("Rays2Polyhedron","outofmem","out of memory space");
02114     return 0;
02115   }
02116   
02117   /* Initialize the constraint matrix 'Mat' */
02118   Vector_Set(Mat->p_Init,0,NbMaxConstrs * (Dimension+1));
02119   for (i=0; i<Dimension; i++) {
02120     
02121     /* Mat->p[i][i+1]=1 */
02122     value_set_si(Mat->p[i][i+1],1);
02123   }
02124   
02125   /* Allocate and assign the saturation matrix. Remember we are using a */
02126   /* transposed saturation matrix referenced by (constraint,ray) pair.  */
02127   Mat->NbRows = Dimension;
02128   nbcolumns = (Ray->NbRows -1)/(sizeof(int)*8) + 1;
02129   SatTranspose = SMAlloc(NbMaxConstrs,nbcolumns);
02130   SMVector_Init(SatTranspose->p[0],Dimension * nbcolumns);

02137   Matrix_Print(stderr,0,Mat);
02138   fprintf(stderr, "\nSatTranspose = ");
02139   SMPrint (SatTranspose);
02140 #endif
02141   
02142   CATCH(any_exception_error) {
02143     
02144     /* In case of overflow, free the allocated memory before forwarding
02145      * the exception. 
02146      */
02147     if (SatTranspose) SMFree(&SatTranspose);
02148     if (Sat) SMFree(&Sat);
02149     if (Mat) Matrix_Free(Mat);
02150     if (Pol) Polyhedron_Free(Pol);
02151     RETHROW();

02153   TRY {
02154

02157     
02158 #ifdef POLY_DEBUG
02159     fprintf(stderr, "[ray2polyhedron: After]\nRay = ");
02160     Matrix_Print(stderr,0,Ray);
02161     fprintf(stderr, "\nConstraints = ");
02162     Matrix_Print(stderr,0,Mat);
02163     fprintf(stderr, "\nSatTranspose = ");
02164     SMPrint (SatTranspose);
02165 #endif
02166     
02167     /* Transform the saturation matrix 'SatTranspose' in the standard  */
02168     /* format, that is, ray X constraint format.                       */
02169     Sat = TransformSat(Mat,Ray,SatTranspose);
02170     
02171 #ifdef POLY_DEBUG
02172     fprintf(stderr, "\nSat =");
02173     SMPrint(Sat);
02174 #endif
02175     
02176     SMFree(&SatTranspose), SatTranspose = NULL;
02177     
02178     /* Remove redundant rays from the ray matrix 'Ray' */
02179     Pol = Remove_Redundants(Mat,Ray,Sat,0);
02180   } /* of TRY */
02181   
02182   UNCATCH(any_exception_error);
02183   
02184 #ifdef POLY_DEBUG
02185   fprintf(stderr, "\nPol = ");
02186   Polyhedron_Print(stderr,"%4d",Pol);
02187 #endif
02188   
02189   SMFree(&Sat);
02190   Matrix_Free(Mat);
02191   return Pol;
02192 } /* Rays2Polyhedron */

02230 /*   
02231  * Build a saturation matrix from constraint matrix 'Mat' and ray matrix 
02232  * 'Ray'. Only 'NbConstraints' constraint of matrix 'Mat' are considered 
02233  * in creating the saturation matrix. 'NbMaxRays' is the maximum number 
02234  * of rows (rays) allowed in the saturation matrix.
02235  * Vin100's stuff, for the polyparam vertices to work.
02236  */
02237 static SatMatrix *BuildSat(Matrix *Mat,Matrix *Ray,unsigned NbConstraints,unsigned NbMaxRays) {
02238   
02239   SatMatrix *Sat = NULL;
02240   int i, j, k, jx;
02241   Value *p1, *p2, *p3;
02242   unsigned Dimension, NbRay, bx, nbcolumns;
02243   
02244   CATCH(any_exception_error) {
02245     if (Sat) 
02246       SMFree(&Sat);
02247     RETHROW();
02248   }
02249   TRY {
02250     NbRay = Ray->NbRows;
02251     Dimension = Mat->NbColumns-1;   /* Homogeneous Dimension */
02252     
02253     /* Build the Sat matrix */
02254     nbcolumns = (Mat->NbRows - 1)/(sizeof(int)*8) + 1;
02255     Sat = SMAlloc(NbMaxRays,nbcolumns);
02256     Sat->NbRows = NbRay;
02257     SMVector_Init(Sat->p_init, nbcolumns * NbRay);
02258     jx=0; bx=MSB;
02259     for (k=0; k<NbConstraints; k++) {
02260       for (i=0; i<NbRay; i++) {
02261         
02262         /* Compute the dot product of ray(i) and constraint(k) and */
02263         /* store in the status element of ray(i).                  */
02264         p1 = Ray->p[i]+1;
02265         p2 = Mat->p[k]+1;
02266         p3 = Ray->p[i];
02267         value_set_si(*p3,0);
02268         for (j=0; j<Dimension; j++) {
02269           value_addmul(*p3, *p1, *p2);
02270           p1++; p2++;
02271         }
02272       }
02273       for (j=0; j<NbRay; j++) {
02274         
02275         /* Set 1 in the saturation matrix if the ray doesn't saturate */
02276         /* the constraint, otherwise the entry is 0.                  */
02277         if (value_notzero_p(Ray->p[j][0]))
02278           Sat->p[j][jx]|=bx;
02279       }
02280       NEXT(jx, bx);
02281     }
02282   } /* end of TRY */
02283   
02284   UNCATCH(any_exception_error);
02285   return Sat;
02286 } /* BuildSat */

02291  * 'NbMaxRays'.   
02292  */
02293 Polyhedron *AddConstraints(Value *Con,unsigned NbConstraints,Polyhedron *Pol,unsigned NbMaxRays) {
02294 
02295   Polyhedron *NewPol = NULL;
02296   Matrix   *Mat = NULL, *Ray = NULL;
02297   SatMatrix *Sat = NULL;
02298   unsigned NbRay, NbCon, Dimension;
02299 
02300   if (NbConstraints == 0)
02301     return Polyhedron_Copy(Pol);
02302   
02303   POL_ENSURE_INEQUALITIES(Pol);
02304   Dimension     = Pol->Dimension + 2;   /* Homogeneous Dimension + Status */
02305 
02306   if (POL_ISSET(NbMaxRays, POL_NO_DUAL)) {
02307     NbCon       = Pol->NbConstraints + NbConstraints;
02308     
02309     Mat = Matrix_Alloc(NbCon, Dimension);
02310     if (!Mat) {
02311       errormsg1("AddConstraints", "outofmem", "out of memory space");
02312       return NULL;
02313     }
02314     Vector_Copy(Pol->Constraint[0], Mat->p[0], Pol->NbConstraints * Dimension);
02315     Vector_Copy(Con, Mat->p[Pol->NbConstraints], NbConstraints * Dimension);  
02316     NewPol = Constraints2Polyhedron(Mat, NbMaxRays);
02317     Matrix_Free(Mat);  
02318     return NewPol;

02322 
02323   CATCH(any_exception_error) {

02325     if (Mat) Matrix_Free(Mat);
02326     if (Ray) Matrix_Free(Ray);
02327     if (Sat) SMFree(&Sat);
02328     RETHROW();

02333 
02334     if (NbRay > NbMaxRays)

02336     
02337     Mat = Matrix_Alloc(NbCon, Dimension);
02338     if(!Mat) {
02339       errormsg1("AddConstraints", "outofmem", "out of memory space");
02340       UNCATCH(any_exception_error);
02341       return 0;
02342     }
02343     
02344     /* Copy constraints of polyhedron 'Pol' to matrix 'Mat' */
02345     Vector_Copy(Pol->Constraint[0], Mat->p[0], Pol->NbConstraints * Dimension);
02346     
02347     /* Add the new constraints pointed by 'Con' to matrix 'Mat' */
02348     Vector_Copy(Con, Mat->p[Pol->NbConstraints], NbConstraints * Dimension);  
02349     
02350     /* Allocate space for ray matrix 'Ray' */
02351     Ray = Matrix_Alloc(NbMaxRays, Dimension);
02352     if(!Ray) {
02353       errormsg1("AddConstraints", "outofmem", "out of memory space");
02354       UNCATCH(any_exception_error);
02355       return 0;
02356     }
02357     Ray->NbRows = NbRay;
02358 
02359     /* Copy rays of polyhedron 'Pol' to matrix 'Ray' */
02360     if (NbRay)
02361         Vector_Copy(Pol->Ray[0], Ray->p[0], NbRay * Dimension);

02382 /* 
02383  * Return 1 if 'Pol1' includes (covers) 'Pol2', 0 otherwise. 
02384  * Polyhedron 'A' includes polyhedron 'B' if the rays of 'B' saturate
02385  * the equalities and verify the inequalities of 'A'. Both 'Pol1' and 
02386  * 'Pol2' have same dimensions. 
02387  */
02388 int PolyhedronIncludes(Polyhedron *Pol1,Polyhedron *Pol2) {
02389         
02390   int Dimension = Pol1->Dimension + 1;   /* Homogenous Dimension */
02391   int i, j, k;
02392   Value *p1, *p2, p3;
02393   
02394   POL_ENSURE_FACETS(Pol1);
02395   POL_ENSURE_VERTICES(Pol1);
02396   POL_ENSURE_FACETS(Pol2);
02397   POL_ENSURE_VERTICES(Pol2);
02398 
02399   value_init(p3); 
02400   for (k=0; k<Pol1->NbConstraints; k++) {
02401     for (i=0;i<Pol2->NbRays;i++) {
02402       
02403       /* Compute the dot product of ray(i) and constraint(k) and store in p3 */
02404       p1 = Pol2->Ray[i]+1;
02405       p2 = Pol1->Constraint[k]+1;
02406       value_set_si(p3,0);
02407       for(j=0;j<Dimension;j++) {
02408         value_addmul(p3, *p1,*p2);
02409         p1++; p2++;
02410       }
02411      
02412       /* If (p3 < 0) or (p3 > 0 and (constraint(k) is equality
02413                                      or ray(i) is a line)), return 0 */
02414       if(value_neg_p(p3) ||
02415           (value_notzero_p(p3)
02416              && (value_zero_p(Pol1->Constraint[k][0]) || (value_zero_p(Pol2->Ray[i][0]))   ) )) {
02417         value_clear(p3); 
02418         return 0;
02419       }
02420     }
02421   } 
02422   value_clear(p3); 
02423   return 1;
02424 } /* PolyhedronIncludes */

02426 /*
02427  * Add Polyhedron 'Pol' to polhedral domain 'PolDomain'. If 'Pol' covers
02428  * some polyhedron in the domain 'PolDomain', it is removed from the list.
02429  * On the other hand if some polyhedron in the domain covers polyhedron 
02430  * 'Pol' then 'Pol' is not included in the domain.   
02431  */
02432 Polyhedron *AddPolyToDomain(Polyhedron *Pol,Polyhedron *PolDomain) {
02433   
02434   Polyhedron *p, *pnext, *p_domain_end = (Polyhedron *) 0;
02435   int Redundant;
02436   
02437   if (!Pol) 
02438     return PolDomain;
02439   if (!PolDomain)       
02440     return Pol;
02441 
02442   POL_ENSURE_FACETS(Pol);
02443   POL_ENSURE_VERTICES(Pol);
02444   
02445   /* Check for emptiness of polyhedron 'Pol' */
02446   if (emptyQ(Pol)) {
02447     Polyhedron_Free(Pol);
02448     return PolDomain;
02449   }
02450   
02451   POL_ENSURE_FACETS(PolDomain);
02452   POL_ENSURE_VERTICES(PolDomain);
02453 
02454   /* Check for emptiness of polyhedral domain 'PolDomain' */
02455   if (emptyQ(PolDomain)) {
02456     Polyhedron_Free(PolDomain);
02457     return Pol;
02458   }
02459   
02460   /* Test 'Pol' against the domain 'PolDomain' */
02461   Redundant = 0;
02462   for (p=PolDomain,PolDomain=(Polyhedron *)0; p; p=pnext) {
02463     
02464     /* If 'Pol' covers 'p' */    
02465     if (PolyhedronIncludes(Pol, p))
02466     {
02467        /* free p */
02468                  pnext = p->next;
02469        Polyhedron_Free( p );
02470        continue;
02471     }
02472 
02473     /* Add polyhedron p to the new domain list */
02474     if (!PolDomain) PolDomain = p; else p_domain_end->next = p;
02475     p_domain_end = p;
02476     
02477     /* If p covers Pol */
02478     if (PolyhedronIncludes(p,Pol)) {
02479       Redundant = 1;
02480       break;
02481     }
02482     pnext = p->next;
02483   }
02484   if (!Redundant) {  
02485     
02486     /* The whole list has been checked. Add new polyhedron 'Pol' to the */
02487     /* new domain list.                                                 */ 
02488     if (!PolDomain) PolDomain = Pol; else p_domain_end->next = Pol;
02489   }
02490   else {
02491     
02492     /* The rest of the list is just inherited from p */
02493     Polyhedron_Free(Pol);
02494   }
02495   return PolDomain;
02496 } /* AddPolyToDomain */

02502  * maximum allowed rays in the new generated polyhedron.

02506  * If Pass == 3, add ( +constraint   ) >= 0
02507  */

02525   }
02526   TRY {
02527     
02528     /* If 'Con' is the positivity constraint, return Null */
02529     Dimension  = Pol->Dimension+1;      /* Homogeneous Dimension */

02552       UNCATCH(any_exception_error);
02553       return 0;
02554     }
02555     
02556     /* Set the constraints of Pol */
02557     Vector_Copy(Pol->Constraint[0], Mat->p[0], NbEle1);

02559     /* Add the new constraint */
02560     value_set_si(Mat->p[NbCon][0],1);

02576     }

02583     /* Create the saturation matrix from the constraint matrix 'mat' and */
02584     /* ray matrix 'Ray'.                                                 */
02585     Sat = BuildSat(Mat, Ray, NbCon, NbMaxRays);

02587     /* Create the ray matrix 'Ray' from consraint matrix 'Mat'           */
02588     Pol_status = Chernikova(Mat, Ray, Sat, Pol->NbBid, NbMaxRays, NbCon,0);

02590     /* Remove redundant constraints from matrix 'Mat' */ 
02591     NewPol = Remove_Redundants(Mat, Ray, Sat, 0);
02592     
02593   } /* end of TRY */
02594   
02595   UNCATCH(any_exception_error);
02596   
02597   SMFree(&Sat);
02598   Matrix_Free(Ray);

02600   return NewPol;
02601 } /* SubConstraint */

02833 /*
02834  * Given constraint number 'k' of a polyhedron, and an array 'Filter' to store
02835  * the non-redundant constraints of the polyhedron in bit-wise notation, and
02836  * a Matrix 'Sat', add the constraint 'k' in 'Filter' array. tmpR[i] stores the
02837  * number of constraints, other than those in 'Filter', which ray(i) saturates 
02838  * or verifies. In case, ray(i) does not saturate or verify a constraint in
02839  * array 'Filter', it is assigned to -1. Similarly, tmpC[j] stores the number
02840  * of rays which constraint(j), if it doesn't belong to Filter, saturates or 
02841  * verifies. If constraint(j) belongs to 'Filter', then tmpC[j] is assigned to
02842  * -1. 'NbConstraints' is the number of constraints in the constraint matrix of
02843  * the polyhedron. 
02844  * NOTE: (1) 'Sat' is not the saturation matrix of the polyhedron. In fact, 
02845  *           entry in 'Sat' is set to 1 if ray(i) of polyhedron1 verifies or 
02846  *           saturates the constraint(j) of polyhedron2 and otherwise it is set
02847  *           to 0. So here the difference with saturation matrix is in terms 
02848  *           definition and entries(1<->0) of the matrix 'Sat'.    
02849  *       
02850  * ALGORITHM:->
02851  * (1) Include constraint(k) in array 'Filter'. 
02852  * (2) Set tmpC[k] to -1.
02853  * (3) For all ray(i) {
02854  *        If ray(i) saturates or verifies constraint(k) then --(tmpR[i])
02855  *        Else {
02856  *           Discard ray(i) by assigning tmpR[i] = -1
02857  *           Decrement tmpC[j] for all constraint(j) not in array 'Filter'.
02858  *        }
02859  *     }
02860  */
02861 static void addToFilter(int k, unsigned *Filter, SatMatrix *Sat,Value *tmpR,Value *tmpC,int NbRays,int NbConstraints) {
02862   
02863   int kj, i,j, jx;
02864   unsigned kb, bx;
02865   
02866   /* Remove constraint k */
02867   kj =   k/WSIZE; kb = MSB; kb >>= k%WSIZE;
02868   Filter[kj]|=kb;
02869   value_set_si(tmpC[k],-1);
02870   
02871   /* Remove rays excluded by constraint k */
02872   for(i=0; i<NbRays; i++)
02873     if (value_posz_p(tmpR[i])) {
02874       if (Sat->p[i][kj]&kb)
02875         value_decrement(tmpR[i],tmpR[i]);  /* adjust included ray */
02876       else {
02877         
02878         /* Constraint k excludes ray i -- delete ray i */
02879         value_set_si(tmpR[i],-1);
02880         
02881         /* Adjust non-deleted constraints */
02882         jx=0; bx=MSB;
02883         for(j=0; j<NbConstraints; j++) {
02884           if (value_posz_p(tmpC[j]) && (Sat->p[i][jx]&bx) )
02885             value_decrement(tmpC[j],tmpC[j]);
02886           NEXT(jx,bx);
02887         }
02888       }
02889     } 
02890 } /* addToFilter */

02892 /*
02893  * Given polyhedra 'P1' and 'P2' such that their intersection is an empty
02894  * polyhedron, find the minimal set of constraints of 'P1' which contradict
02895  * all of the constraints of 'P2'. This is believed to be an NP-hard problem
02896  * and so a heuristic is employed to solve it in worst case. The heuristic is 
02897  * to select in every turn that constraint of 'P1' which excludes most rays of
02898  * 'P2'. A bit in the binary format of an element of array 'Filter' is set to
02899  * 1 if the corresponding constraint is to be included in the minimal set of 
02900  * constraints otherwise it is set to 0.
02901  */
02902 static void FindSimple(Polyhedron *P1,Polyhedron *P2,unsigned *Filter,unsigned NbMaxRays) {
02903   
02904   Matrix *Mat = NULL;
02905   SatMatrix *Sat = NULL;
02906   int i, j, k, jx, found;
02907   Value *p1, *p2, p3;
02908   unsigned Dimension, NbRays, NbConstraints, bx, nc;
02909   Value NbConstraintsLeft, tmp;
02910   Value *tmpC = NULL, *tmpR = NULL;
02911   Polyhedron *Pol = NULL, *Pol2 = NULL;
02912   
02913   /* Initialize all the 'Value' variables */
02914   value_init(p3); value_init(NbConstraintsLeft);
02915   value_init(tmp);
02916  
02917   CATCH(any_exception_error) {
02918     if (tmpC) free(tmpC);
02919     if (tmpR) free(tmpR);
02920     if (Mat) Matrix_Free(Mat);
02921     if (Sat) SMFree(&Sat);
02922     if (Pol2 && Pol2!=P2) Polyhedron_Free(Pol2);
02923     if (Pol && Pol!=Pol2 && Pol!=P2) Polyhedron_Free(Pol);
02924     
02925     /* Clear all the 'Value' variables */
02926     value_clear(p3); value_clear(NbConstraintsLeft);
02927     value_clear(tmp);
02928     RETHROW();
02929   }
02930   TRY {
02931     
02932     Dimension = P1->Dimension+2;       /* status + homogeneous Dimension */
02933     Mat = Matrix_Alloc(P1->NbConstraints, Dimension);
02934     if(!Mat) {
02935       errormsg1("FindSimple", "outofmem", "out of memory space");
02936       UNCATCH(any_exception_error);
02937       
02938       /* Clear all the 'Value' variables */

03281   NbCon = 0;

03300     /* Allocate space for array 'Filter' */

03605  * Given polyhedral domains 'Pol1' and 'Pol2', create a new polyhedral 
03606  * domain which is mathematically the differnce of the two domains. 
03607  */
03608 Polyhedron *DomainDifference(Polyhedron *Pol1,Polyhedron *Pol2,unsigned NbMaxRays) {
03609 
03610   Polyhedron *p1, *p2, *p3, *d;
03611   int i;
03612   
03613   if (!Pol1 || !Pol2) return (Polyhedron*) 0;
03614   if (Pol1->Dimension != Pol2->Dimension) {
03615     errormsg1("DomainDifference", 
03616               "diffdim", "operation on different dimensions");
03617     return (Polyhedron*) 0;
03618   }
03619   POL_ENSURE_FACETS(Pol1);
03620   POL_ENSURE_VERTICES(Pol1);
03621   POL_ENSURE_FACETS(Pol2);
03622   POL_ENSURE_VERTICES(Pol2);
03623   if (emptyQ(Pol1) || emptyQ(Pol2))
03624     return (Domain_Copy(Pol1));
03625   d = (Polyhedron *)0;
03626   for (p2=Pol2; p2; p2=p2->next) {
03627     for (p1=Pol1; p1; p1=p1->next) {
03628       for (i=0; i<p2->NbConstraints; i++) {
03629         
03630         /* Add the constraint ( -p2->constraint[i] -1) >= 0 in 'p1' */
03631         /* and create the new polyhedron 'p3'.                      */
03632         p3 = SubConstraint(p2->Constraint[i], p1, NbMaxRays,0);
03633         
03634         /* Add 'p3' in the new domain 'd' */
03635         d = AddPolyToDomain (p3, d);
03636         
03637         /* If the constraint p2->constraint[i][0] is an equality, then  */
03638         /* add the constraint ( +p2->constraint[i] -1) >= 0  in 'p1' and*/
03639         /* create the new polyhedron 'p3'.                              */
03640         
03641         if( value_notzero_p(p2->Constraint[i][0]) ) /* Inequality */
03642           continue;  
03643         p3 = SubConstraint(p2->Constraint[i], p1, NbMaxRays,1);
03644         
03645         /* Add 'p3' in the new domain 'd' */
03646         d = AddPolyToDomain (p3, d);
03647       }
03648     }
03649     if (p2 != Pol2)
03650         Domain_Free(Pol1);
03651     Pol1 = d;
03652     d = (Polyhedron *)0;
03653   }
03654   if (!Pol1)
03655     return Empty_Polyhedron(Pol2->Dimension);
03656   else
03657     return Pol1;

03661  * Given a polyhedral domain 'Pol', convert it to a new polyhedral domain 
03662  * with dimension expanded to 'align_dimension'. 'NbMaxRays' is the maximum
03663  * allowed rays in the new polyhedra.
03664  */

03667   int i, j, k;
03668   Polyhedron *p = NULL, **next, *result = NULL;

03672     if (result) Polyhedron_Free(result);
03673     RETHROW();

03675   TRY {
03676     
03677     if (!Pol) return Pol;
03678     dim = Pol->Dimension;
03679     if (align_dimension < Pol->Dimension) {
03680       errormsg1("align_context", "diffdim", "context dimension exceeds data");
03681       UNCATCH(any_exception_error);

03685       UNCATCH(any_exception_error);

03690     k = align_dimension - Pol->Dimension;

03694     for (; Pol; Pol=Pol->next) {

03708       if (have_cons) {
03709         for (i = 0; i < NbCons; ++i) {
03710           value_assign(p->Constraint[i][0], Pol->Constraint[i][0]);  /* Status bit */
03711           Vector_Copy(Pol->Constraint[i]+1, p->Constraint[i]+k+1, Pol->Dimension+1);
03712         }
03713         p->NbEq = Pol->NbEq;
03714       }
03715 
03716       if (have_rays) {
03717         for (i = 0; i < k; ++i)
03718           value_set_si(p->Ray[i][1+i], 1);                          /* A line */
03719         for (i = 0; i < Pol->NbRays; ++i) {
03720           value_assign(p->Ray[k+i][0], Pol->Ray[i][0]);             /* Status bit */
03721           Vector_Copy(Pol->Ray[i]+1, p->Ray[i+k]+k+1, Pol->Dimension+1);
03722         }
03723         p->NbBid = Pol->NbBid + k;
03724       }
03725       p->flags = Pol->flags;
03726       
03727       *next = p;
03728       next = &p->next;
03729     }
03730   } /* end of TRY */
03731   
03732   UNCATCH(any_exception_error);

03738 /*       D : Domain to be scanned (single polyhedron only)              */
03739 /*       C : Context domain                                             */
03740 /*       NbMaxRays : Workspace size                                     */
03741 /* Returns a linked list of scan domains, outer loop first              */
03742 /*----------------------------------------------------------------------*/
03743 Polyhedron *Polyhedron_Scan(Polyhedron *D, Polyhedron *C,unsigned NbMaxRays) {
03744   
03745   int i, j, dim ;

03751   res = last = (Polyhedron *) 0;
03752   if (dim==0) return (Polyhedron *)0;
03753 
03754   assert(!D->next);
03755 
03756   POL_ENSURE_FACETS(D);
03757   POL_ENSURE_VERTICES(D);
03758   POL_ENSURE_FACETS(C);
03759   POL_ENSURE_VERTICES(C);
03760 
03761   /* Allocate space for constraint matrix. */
03762   Mat   = Matrix_Alloc(D->Dimension, D->Dimension+2);
03763   if(!Mat) {
03764     errormsg1("Polyhedron_Scan", "outofmem", "out of memory space");
03765     return 0;
03766   }
03767   C1  = align_context(C,D->Dimension,NbMaxRays);
03768   if(!C1) {
03769     return 0;
03770   }
03771   /* Vin100, aug 16, 2001:  The context is intersected with D */

03789   }
03790   Domain_Free(D2);
03791   Domain_Free(C1);

03797 /* int lower_upper_bounds(pos,P,context,LBp,UBp)                       */
03798 /*    pos : index position of current loop index (1..hdim-1)           */
03799 /*    P: loop domain                                                   */
03800 /*    context : context values for fixed indices                       */
03801 /*              notice that context[hdim] must be 1                    */
03802 /*    LBp, UBp : pointers to resulting bounds                          */
03803 /* returns the flag = (UB_INFINITY, LB_INFINITY)                       */
03804 /*---------------------------------------------------------------------*/
03805 int lower_upper_bounds(int pos,Polyhedron *P,Value *context,Value *LBp,Value *UBp) {
03806   
03807   Value LB, UB;
03808   int flag, i;
03809   Value n, n1, d, tmp;
03810   
03811   POL_ENSURE_FACETS(P);
03812   POL_ENSURE_VERTICES(P);
03813 
03814   /* Initialize all the 'Value' variables */
03815   value_init(LB); value_init(UB); value_init(tmp);
03816   value_init(n); value_init(n1); value_init(d);
03817   
03818   value_set_si(LB,0);
03819   value_set_si(UB,0);
03820   
03821   /* Compute Upper Bound and Lower Bound for current loop */
03822   flag = LB_INFINITY | UB_INFINITY;
03823   for (i=0; i<P->NbConstraints; i++) {
03824     value_assign(d,P->Constraint[i][pos]);
03825     Inner_Product(&context[1],&(P->Constraint[i][1]),P->Dimension+1,&n);
03826     if (value_zero_p(d)) {
03827       /* If context doesn't satisfy constraints, return empty loop. */
03828       if (value_neg_p(n) ||
03829           (value_zero_p(P->Constraint[i][0]) && value_notzero_p(n)))
03830         goto empty_loop;
03831       continue;
03832     }
03833     value_oppose(n,n);
03834     
03835     /*---------------------------------------------------*/
03836     /* Compute n/d        n/d<0              n/d>0       */
03837     /*---------------------------------------------------*/
03838     /*  n%d == 0    floor   = n/d      floor   = n/d     */
03839     /*              ceiling = n/d      ceiling = n/d     */
03840     /*---------------------------------------------------*/
03841     /*  n%d != 0    floor   = n/d - 1  floor   = n/d     */
03842     /*              ceiling = n/d      ceiling = n/d + 1 */
03843     /*---------------------------------------------------*/
03844 
03845     /* Check to see if constraint is inequality */
03846     /* if constraint is equality, both upper and lower bounds are fixed */
03847     if(value_zero_p(P->Constraint[i][0])) {     /* Equality */
03848       value_modulus(tmp,n,d);
03849       
03850       /* if not integer, return 0; */
03851       if (value_notzero_p(tmp))
03852         goto empty_loop;
03853 
03854       value_division(n1,n,d);
03855       
03856       /* Upper and Lower bounds found */
03857       if((flag&LB_INFINITY) || value_gt(n1,LB))
03858         value_assign(LB,n1);
03859       if((flag&UB_INFINITY) || value_lt(n1,UB))
03860         value_assign(UB,n1);
03861       flag = 0;
03862     }
03863     
03864     if (value_pos_p(d)) {  /* Lower Bound */
03865       value_modulus(tmp,n,d);
03866       
03867       /* n1 = ceiling(n/d) */
03868       if (value_pos_p(n) && value_notzero_p(tmp)) {
03869         value_division(n1,n,d);
03870         value_add_int(n1,n1,1);
03871       }
03872       else
03873         value_division(n1,n,d);
03874       if (flag&LB_INFINITY) {
03875         value_assign(LB,n1); 
03876         flag^=LB_INFINITY; 
03877       }
03878       else if(value_gt(n1,LB))
03879         value_assign(LB,n1);
03880     }
03881     
03882     if (value_neg_p(d)) {   /* Upper Bound */
03883       value_modulus(tmp,n,d);
03884       
03885       /* n1 = floor(n/d) */
03886       if (value_pos_p(n) && value_notzero_p(tmp)) {
03887         value_division(n1,n,d);
03888         value_sub_int(n1,n1,1);
03889       }
03890       else
03891         value_division(n1,n,d);
03892       
03893       if (flag&UB_INFINITY) {
03894         value_assign(UB,n1); 
03895         flag^=UB_INFINITY; 
03896       }
03897       else if (value_lt(n1,UB))
03898         value_assign(UB, n1);
03899     }
03900   }
03901   if ((flag & LB_INFINITY)==0) value_assign(*LBp,LB);
03902   if ((flag & UB_INFINITY)==0) value_assign(*UBp,UB);
03903 
03904   if (0) {
03905 empty_loop:
03906     flag = 0;
03907     value_set_si(*LBp, 1);
03908     value_set_si(*UBp, 0);      /* empty loop */
03909   }
03910   
03911   /* Clear all the 'Value' variables */
03912   value_clear(LB); value_clear(UB); value_clear(tmp);
03913   value_clear(n); value_clear(n1); value_clear(d);
03914   return flag;
03915 } /* lower_upper_bounds */

03917 /*
03918  *  C = A x B
03919  */
03920 static void Rays_Mult(Value **A, Matrix *B, Value **C, unsigned NbRays)
03921 {
03922   int i, j, k;
03923   unsigned Dimension1, Dimension2;
03924   Value Sum, tmp;
03925 
03926   value_init(Sum); value_init(tmp);
03927 
03928   CATCH(any_exception_error) {
03929     value_clear(Sum); value_clear(tmp);
03930     RETHROW();
03931   }
03932   TRY {
03933     Dimension1 = B->NbRows;
03934     Dimension2 = B->NbColumns;
03935 
03936     for (i=0; i<NbRays; i++) {
03937       value_assign(C[i][0],A[i][0]);
03938       for (j=0; j<Dimension2; j++) {
03939         value_set_si(Sum,0);
03940         for (k=0; k<Dimension1; k++) {
03941           
03942           /* Sum+=A[i][k+1] * B->p[k][j]; */

03944         }
03945         value_assign(C[i][j+1],Sum);
03946       }
03947       Vector_Gcd(C[i]+1, Dimension2, &tmp);
03948       if (value_notone_p(tmp))
03949           Vector_AntiScale(C[i]+1, C[i]+1, tmp, Dimension2);
03950     }
03951   }
03952   UNCATCH(any_exception_error);
03953   value_clear(Sum); value_clear(tmp);

04118     i   |___|    |   j  |___| |          i   |___|
04119      Pol->Rays  \       Func /               Rays
04120 
04121     */
04122 
04123     if (Dimension1 == Dimension2) {
04124         Matrix *M, *M2;
04125         int ok;
04126         M = Matrix_Copy(Func);
04127         M2 = Matrix_Alloc(Dimension2, Dimension1);
04128         if (!M2) {
04129           errormsg1("Polyhedron_Image", "outofmem", "out of memory space\n");
04130           UNCATCH(any_exception_error);
04131           return 0;
04132         }
04133 
04134         ok = Matrix_Inverse(M, M2);
04135         Matrix_Free(M);
04136         if (ok) {
04137             NewPol = Polyhedron_Alloc(Pol->Dimension, Pol->NbConstraints,
04138                                       Pol->NbRays);
04139             if (!NewPol) {
04140               errormsg1("Polyhedron_Image", "outofmem", 
04141                         "out of memory space\n");
04142               UNCATCH(any_exception_error);
04143               return 0;
04144             }
04145             Rays_Mult_Transpose(Pol->Ray, Func, NewPol->Ray, NbRays);
04146             Rays_Mult(Pol->Constraint, M2, NewPol->Constraint, 
04147                       Pol->NbConstraints);
04148             NewPol->NbEq = Pol->NbEq;
04149             NewPol->NbBid = Pol->NbBid;
04150             if (NewPol->NbEq)
04151               Gauss4(NewPol->Constraint, NewPol->NbEq, NewPol->NbConstraints,
04152                      NewPol->Dimension+1);
04153             if (NewPol->NbBid)
04154               Gauss4(NewPol->Ray, NewPol->NbBid, NewPol->NbRays,
04155                      NewPol->Dimension+1);
04156         }
04157         Matrix_Free(M2);
04158     }
04159     
04160     if (!NewPol) {
04161         /* Allocate space for the resulting ray matrix */
04162         Rays = Matrix_Alloc(NbRays, Dimension2+1);
04163         if (!Rays) {
04164           errormsg1("Polyhedron_Image", "outofmem", "out of memory space\n");
04165           UNCATCH(any_exception_error);
04166           return 0;
04167         }
04168         
04169         /* The new ray space is the product of ray matrix of the polyhedron and */
04170         /* the transpose matrix of the mapping function.                        */
04171         Rays_Mult_Transpose(Pol->Ray, Func, Rays->p, NbRays);
04172         NewPol = Rays2Polyhedron(Rays, NbMaxConstrs);
04173         Matrix_Free(Rays), Rays = NULL;
04174     }
04175     
04176   } /* end of TRY */
04177 
04178   UNCATCH(any_exception_error);
04179   return NewPol;

04213 /* 
04214  * Given a polyhedron 'Pol' and an affine cost function 'Cost', compute the 
04215  * maximum and minimum value of the function over set of points representing
04216  * polyhedron. 
04217  * Note: If Polyhedron 'Pol' is empty, then there is no feasible solution. 
04218  * Otherwise, if there is a bidirectional ray with Sum[cost(i)*ray(i)] != 0 or
04219  * a unidirectional ray with Sum[cost(i)*ray(i)] >0, then the maximum is un-
04220  * bounded else the finite optimal solution occurs at one of the vertices of
04221  * the polyhderon. 
04222  */
04223 Interval *DomainCost(Polyhedron *Pol,Value *Cost) {
04224   
04225   int i, j, NbRay, Dim;
04226   Value *p1, *p2, p3, d, status;
04227   Value tmp1, tmp2, tmp3;
04228   Value **Ray;
04229   Interval *I = NULL;
04230 
04231   value_init(p3); value_init(d); value_init(status);
04232   value_init(tmp1); value_init(tmp2); value_init(tmp3);
04233 
04234   POL_ENSURE_FACETS(Pol);
04235   POL_ENSURE_VERTICES(Pol);
04236 
04237   CATCH(any_exception_error) {
04238     if (I) free(I);
04239     RETHROW();
04240     value_clear(p3); value_clear(d); value_clear(status);
04241     value_clear(tmp1); value_clear(tmp2); value_clear(tmp3);
04242   }
04243   TRY {
04244     
04245     Ray = Pol->Ray;
04246     NbRay = Pol->NbRays;
04247     Dim = Pol->Dimension+1;             /* Homogenous Dimension */
04248     I = (Interval *) malloc (sizeof(Interval));
04249     if (!I) {
04250       errormsg1("DomainCost", "outofmem", "out of memory space\n");
04251       UNCATCH(any_exception_error);
04252       value_clear(p3); value_clear(d); value_clear(status);
04253       value_clear(tmp1); value_clear(tmp2); value_clear(tmp3);
04254       return 0;
04255     }
04256     
04257     /* The maximum and minimum values of the cost function over polyhedral  */
04258     /* domain is stored in 'I'. I->MaxN and I->MaxD store the numerator and */
04259     /* denominator of the maximum value. Likewise,I->MinN and I->MinD store */
04260     /* the numerator and denominator of the minimum value. I->MaxI and      */
04261     /* I->MinI store the ray indices corresponding to the max and min values*/
04262     /* of the function.                                                     */
04263     
04264     value_set_si(I->MaxN,-1);
04265     value_set_si(I->MaxD,0);          /* Actual cost is MaxN/MaxD */
04266     I->MaxI = -1;
04267     value_set_si(I->MinN,1);
04268     value_set_si(I->MinD,0);
04269     I->MinI = -1;
04270     
04271     /* Compute the cost of each ray[i] */
04272     for (i=0; i<NbRay; i++) {
04273       p1 = Ray[i];
04274       value_assign(status, *p1);
04275       p1++;
04276       p2 = Cost;
04277       
04278       /* p3 = *p1++ * *p2++; */
04279       value_multiply(p3,*p1,*p2);
04280       p1++; p2++;
04281       for (j=1; j<Dim; j++) {
04282         value_multiply(tmp1,*p1,*p2);
04283         
04284         /* p3 += *p1++ * *p2++; */
04285         value_addto(p3,p3,tmp1);
04286         p1++; p2++;
04287       }
04288       
04289       /* d = *--p1; */
04290       p1--;
04291       value_assign(d,*p1); /* d == 0 for lines and ray, non-zero for vertex */
04292       value_multiply(tmp1,p3,I->MaxD); 
04293       value_multiply(tmp2,I->MaxN,d);
04294       value_set_si(tmp3,1);
04295       
04296       /* Compare p3/d with MaxN/MaxD to assign new maximum cost value */
04297       if (I->MaxI==-1 ||
04298           value_gt(tmp1,tmp2) ||
04299           (value_eq(tmp1,tmp2) &&
04300            value_eq(d,tmp3) && value_ne(I->MaxD,tmp3))) {
04301         value_assign(I->MaxN,p3);
04302         value_assign(I->MaxD,d);
04303         I->MaxI = i;
04304       }
04305       value_multiply(tmp1,p3,I->MinD);
04306       value_multiply(tmp2,I->MinN,d);
04307       value_set_si(tmp3,1);
04308       
04309       /* Compare p3/d with MinN/MinD to assign new minimum cost value */
04310       if (I->MinI==-1 ||
04311           value_lt(tmp1,tmp2) ||
04312           (value_eq(tmp1,tmp2) &&
04313            value_eq(d,tmp3) && value_ne(I->MinD,tmp3))) {
04314         value_assign(I->MinN, p3);
04315         value_assign(I->MinD, d);
04316         I->MinI = i;
04317       }
04318       value_multiply(tmp1,p3,I->MaxD);
04319       value_set_si(tmp2,0);
04320       
04321       /* If there is a line, assign max to +infinity and min to -infinity */
04322       if (value_zero_p(status)) { /* line , d is 0 */
04323         if (value_lt(tmp1,tmp2)) {
04324           value_oppose(I->MaxN,p3);
04325           value_set_si(I->MaxD,0);
04326           I->MaxI = i;
04327         }
04328         value_multiply(tmp1,p3,I->MinD);
04329         value_set_si(tmp2,0);
04330 
04331         if (value_gt(tmp1,tmp2)) {
04332           value_oppose(I->MinN,p3);
04333           value_set_si(I->MinD,0);
04334           I->MinI = i;
04335         }
04336       }
04337     }
04338   } /* end of TRY */
04339   
04340   UNCATCH(any_exception_error);
04341   value_clear(p3); value_clear(d); value_clear(status);
04342   value_clear(tmp1); value_clear(tmp2); value_clear(tmp3);
04343   return I;
04344 } /* DomainCost */

04394 /* 
04395  * Computes the disjoint union of a union of polyhedra.
04396  * If flag = 0 the result is such that there are no intersections
04397  *                   between the resulting polyhedra,
04398  * if flag = 1 it computes a joint union, the resulting polyhedra are
04399  *                   adjacent (they have their facets in common).
04400  *
04401  * WARNING: if all polyhedra are not of same geometrical dimension
04402  *          duplicates may appear.
04403  */
04404 Polyhedron *Disjoint_Domain( Polyhedron *P, int flag, unsigned NbMaxRays )
04405 {
04406         Polyhedron *lP, *tmp, *Result, *lR, *prec, *reste;
04407         Polyhedron *p1, *p2, *p3, *Pol1, *dx, *d1, *d2, *pi, *newpi;
04408         int i;
04409 
04410         if( flag!=0 && flag!=1 )
04411         {
04412                 errormsg1("Disjoint_Domain",
04413                         "invalidarg", "flag should be equal to 0 or 1");
04414                 return (Polyhedron*) 0;
04415         }
04416         if(!P) return (Polyhedron*) 0;
04417         if(!P->next) return Polyhedron_Copy(P);
04418 
04419         Result = (Polyhedron *)0;
04420 
04421         for(lP=P;lP;lP=lP->next)
04422         {
04423                 reste = Polyhedron_Copy(lP);
04424                 prec = (Polyhedron *)0; /* preceeding lR */
04425                 /* Intersection with each polyhedron of the current Result */
04426                 lR=Result;
04427                 while( lR && reste )
04428                 {
04429                         /* dx = DomainIntersection(reste,lR->P,WS); */
04430                         dx = (Polyhedron *)0;
04431                         for( p1=reste; p1; p1=p1->next )
04432                         {
04433                                 p3 = AddConstraints(lR->Constraint[0], lR->NbConstraints, p1,
04434                                                 NbMaxRays);
04435                                 dx = AddPolyToDomain(p3,dx);
04436                         }
04437 
04438                         /* if empty intersection, continue */
04439                         if(!dx)
04440                         {       prec = lR;
04441                                 lR=lR->next;
04442                                 continue;
04443                         }
04444                         if (emptyQ(dx)) {       
04445                                 Domain_Free(dx);
04446                                 prec = lR;
04447                                 lR=lR->next;
04448                                 continue;
04449                 }
04450 
04451                         /* intersection is not empty, we need to compute the differences */
04452                         /* between the intersection and the two polyhedra, such that the */
04453                         /* results are disjoint unions (according to flag)               */
04454                         /* d1 = reste \ P = DomainDifference(reste,lR->P,WS);   */
04455                         /* d2 = P \ reste = DomainDifference(lR->P,reste,WS); */
04456 
04457                         /* compute d1 */
04458                         d1 = (Polyhedron *)0;
04459                         for (p1=reste; p1; p1=p1->next)
04460                         {
04461                                 pi = p1;
04462                                 for (i=0; i<P->NbConstraints && pi ; i++)
04463                                 {
04464 
04465                                         /* Add the constraint ( -P->constraint[i] [-1 if flag=0]) >= 0 in 'p1' */
04466                                         /* and create the new polyhedron 'p3'.                      */
04467                                         p3 = SubConstraint(P->Constraint[i], pi, NbMaxRays,2*flag);
04468                                         /* Add 'p3' in the new domain 'd1' */
04469                                         d1 = AddPolyToDomain (p3, d1);
04470 
04471                                         /* If the constraint P->constraint[i][0] is an equality, then add   */
04472                                         /* the constraint ( +P->constraint[i] [-1 if flag=0]) >= 0  in 'pi' */
04473                                         /* and create the new polyhedron 'p3'.                              */
04474                                         if( value_zero_p(P->Constraint[i][0]) ) /* Inequality */
04475                                         {
04476                                                 p3 = SubConstraint(P->Constraint[i], pi, NbMaxRays,1+2*flag);
04477                                                 /* Add 'p3' in the new domain 'd1' */
04478                                                 d1 = AddPolyToDomain (p3, d1);
04479 
04480                                                 /* newpi : add constraint P->constraint[i]==0 to pi */
04481                                                 newpi = AddConstraints( P->Constraint[i], 1, pi, NbMaxRays);
04482                                         }
04483                                         else
04484                                         {
04485                                                 /* newpi : add constraint +P->constraint[i] >= 0 in pi */
04486                                                 newpi = SubConstraint(P->Constraint[i], pi, NbMaxRays,3);
04487                                         }
04488                                         if( newpi && emptyQ( newpi ) )
04489                                         {
04490                                                 Domain_Free( newpi );
04491                                                 newpi = (Polyhedron *)0;
04492                                         }
04493                                         if( pi != p1 )
04494                                                 Domain_Free( pi );
04495                                         pi = newpi;
04496                                 }
04497                                 if( pi != p1 )
04498                                         Domain_Free( pi );
04499                         }
04500 
04501                         /* and now d2 */
04502                         Pol1 = Polyhedron_Copy( lR );
04503                         for (p2=reste; p2; p2=p2->next)
04504                         {
04505                                 d2 = (Polyhedron *)0;
04506                                 for (p1=Pol1; p1; p1=p1->next)
04507                                 {
04508                                         pi = p1;
04509                                         for (i=0; i<p2->NbConstraints && pi ; i++)
04510                                         {
04511 
04512                                                 /* Add the constraint ( -p2->constraint[i] [-1 if flag=0]) >= 0 in 'pi' */
04513                                                 /* and create the new polyhedron 'p3'.                      */
04514                                                 p3 = SubConstraint(p2->Constraint[i], pi, NbMaxRays,2*flag);
04515                                                 /* Add 'p3' in the new domain 'd2' */
04516                                                 d2 = AddPolyToDomain (p3, d2);
04517 
04518                                                 /* If the constraint p2->constraint[i][0] is an equality, then add   */
04519                                                 /* the constraint ( +p2->constraint[i] [-1 if flag=0]) >= 0  in 'pi' */
04520                                                 /* and create the new polyhedron 'p3'.                              */
04521                                                 if( value_zero_p(p2->Constraint[i][0]) ) /* Inequality */
04522                                                 {
04523                                                         p3 = SubConstraint(p2->Constraint[i], pi, NbMaxRays,1+2*flag);
04524                                                         /* Add 'p3' in the new domain 'd2' */
04525                                                         d2 = AddPolyToDomain (p3, d2);
04526 
04527                                                         /* newpi : add constraint p2->constraint[i]==0 to pi */
04528                                                         newpi = AddConstraints( p2->Constraint[i], 1, pi, NbMaxRays);
04529                                                 }
04530                                                 else
04531                                                 {
04532                                                         /* newpi : add constraint +p2->constraint[i] >= 0 in pi */
04533                                                         newpi = SubConstraint(p2->Constraint[i], pi, NbMaxRays,3);
04534                                                 }
04535                                                 if( newpi && emptyQ( newpi ) )
04536                                                 {
04537                                                         Domain_Free( newpi );
04538                                                         newpi = (Polyhedron *)0;
04539                                                 }
04540                                                 if( pi != p1 )
04541                                                         Domain_Free( pi );
04542                                                 pi = newpi;
04543                                         }
04544                                         if( pi && pi!=p1 )
04545                                                 Domain_Free( pi );
04546                                 }
04547                                 if( Pol1 )
04548                                         Domain_Free( Pol1 );
04549                                 Pol1 = d2;
04550                         }
04551                         /* ok, d1 and d2 are computed */
04552 
04553                         /* now, replace lR by d2+dx (at least dx is nonempty) and set reste to d1 */
04554                         if( d1 && emptyQ(d1) )
04555                         {
04556                                 Domain_Free( d1 );
04557                                 d1 = NULL;
04558                         }
04559                         if( d2 && emptyQ(d2) )
04560                         {
04561                                 Domain_Free( d2 );
04562                                 d2 = NULL;
04563                         }
04564 
04565                         /* set reste */
04566                         Domain_Free( reste );
04567                         reste = d1;
04568 
04569                         /* add d2 at beginning of Result */
04570                         if( d2 )
04571                         {
04572                                 for( tmp=d2 ; tmp->next ; tmp=tmp->next )
04573                                                 ;
04574                                 tmp->next = Result;
04575                                 Result = d2;
04576                                 if( !prec )
04577                                         prec = tmp;
04578                         }
04579 
04580                         /* add dx at beginning of Result */
04581                         for( tmp=dx ; tmp->next ; tmp=tmp->next )
04582                                 ;
04583                         tmp->next = Result;
04584                         Result = dx;
04585                         if( !prec )
04586                                 prec = tmp;
04587 
04588                         /* suppress current lR */
04589                         if( !prec )
04590                                 errormsg1( "Disjoint_Domain","internalerror","internal error");
04591                         prec->next = lR->next;
04592                         Polyhedron_Free( lR );
04593                         lR = prec->next;
04594                 } /* end for result */
04595 
04596                   /* if there is something left, add it to Result : */
04597                 if(reste)
04598                 {
04599                         if(emptyQ(reste))
04600                         {
04601                                 Domain_Free( reste );
04602                                 reste = NULL;
04603                         }
04604                         else
04605                         {
04606                                 Polyhedron *tnext;
04607                                 for( tmp=reste ; tmp ; tmp=tnext )
04608                                 {
04609                                         tnext = tmp->next;
04610                                         tmp->next = NULL;
04611                                         Result = AddPolyToDomain(tmp, Result);
04612                                 }
04613                 }
04614                 }
04615         }
04616 
04617         return( Result );
04618 }

04679 /*
04680  * Replaces constraint a x >= c by x >= ceil(c/a)
04681  * where "a" is a common factor in the coefficients
04682  * Destroys P and returns a newly allocated Polyhedron
04683  * or just returns P in case no changes were made
04684  */
04685 Polyhedron *DomainConstraintSimplify(Polyhedron *P, unsigned MaxRays)
04686 {
04687     Polyhedron **prev;
04688     int len = P->Dimension+2;
04689     Vector *row = Vector_Alloc(len);
04690     Value g;
04691     Polyhedron *R = P, *N;
04692     value_set_si(row->p[0], 1);
04693     value_init(g);
04694 
04695     for (prev = &R; P; P = N) {
04696         Polyhedron *T;
04697         N = P->next;
04698         T = p_simplify_constraints(P, row, &g, MaxRays);
04699 
04700         if (emptyQ(T) && prev != &R) {
04701             Polyhedron_Free(T);
04702             *prev = NULL;
04703             continue;
04704         }
04705 
04706         if (T != P)
04707             T->next = N;
