00646         else
00647           break;
00648       }
00649       else {

00654           if (value_lt(temp,*left_son)) {
00655             value_assign(*current_node,*left_son);
00656             j=(j<<1)+1;
00657           }
00658           else
00659             break;
00660         }
00661         else {
00662           if (value_lt(temp,*right_son)) {
00663             value_assign(*current_node,*right_son );

00665           }
00666           else

00669       }
