00068 /* MSB, TOP, and NEXT are defined over integer type, not on value type */
00069 /* Put a one in the most significant bit of an int (portable) */
00070 #define MSB ((unsigned)(((unsigned)1)<<(sizeof(int)*8-1)))
00071 
00072 /* Largest representable positive number */
00073 #define TOP ((int)(MSB-1))
00074 
00075 /* Right shift the one bit in b and increment j if the last bit in b is one */
00076 #define NEXT(j,b) { if (!((b)>>=1)) { (b)=MSB; (j)++; } }
