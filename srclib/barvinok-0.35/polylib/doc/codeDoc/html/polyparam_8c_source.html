00057  * Return the intersection of two polyhedral domains 'Pol1' and 'Pol2' such 
00058  * that if the intersection is a polyhedron of lower dimension (a degenerate
00059  * polyhedron) than the operands, it is discarded from the resulting polyhedra

00068     fprintf(stderr,
00069             "? PDomainIntersection: operation on different dimensions\n");

00071   }
00072   
00073   POL_ENSURE_FACETS(Pol1);
00074   POL_ENSURE_VERTICES(Pol1);
00075   POL_ENSURE_FACETS(Pol2);
00076   POL_ENSURE_VERTICES(Pol2);
00077  
00078   d = (Polyhedron *)0;
00079   for (p1=Pol1; p1; p1=p1->next) {
00080     for (p2=Pol2; p2; p2=p2->next) {
00081       p3 = AddConstraints(p2->Constraint[0],
00082                           p2->NbConstraints,p1,NbMaxRays);

00151 static int TestRank(Matrix *Mat) {
00152   
00153   int i,j,k;
00154   Value m1,m2,m3,gcd,tmp;
00155 
00156   /* Initialize all the 'Value' variables */
00157   value_init(m1); value_init(m2); 
00158   value_init(m3); value_init(gcd); value_init(tmp);
00159   
00160   for(k=0;k<Mat->NbColumns;++k) {
00161     
00162     /* If the digonal entry (k,k) is zero, search down the column(k) */
00163     /* starting from row(k) to find a non-zero entry                 */
00164     if(value_zero_p(Mat->p[k][k])) {
00165       for(j=k+1;j<Mat->NbRows;++j) {
00166         
00167         /* If a non-zero entry (j,k) is found */
00168         if(value_notzero_p(Mat->p[j][k])) {
00169           
00170           /* Exchange row(k) and row(j) */
00171           for(i=k;i<Mat->NbColumns;++i) {
00172             value_assign(tmp,Mat->p[j][i]);
00173             value_assign(Mat->p[j][i],Mat->p[k][i]);
00174             value_assign(Mat->p[k][i],tmp);
00175           }
00176           break;
00177         }
00178       }
00179       
00180       /* If no non-zero entry is found then the matrix 'Mat' is not full */
00181       /* ranked. Return zero.                                            */
00182       if(j>=Mat->NbRows) {
00183         
00184         /* Clear all the 'Value' variables */
00185         value_clear(m1); value_clear(m2); 
00186         value_clear(m3); value_clear(gcd); value_clear(tmp);
00187         return 0;
00188       } 
00189     }
00190     
00191     /* Now Mat[k][k] is the pivot element */
00192     for(j=k+1;j<Mat->NbRows;++j) {
00193       
00194       /* Make every other entry (below row(k)) in column(k) zero */
00195       value_gcd(gcd, Mat->p[j][k], Mat->p[k][k]);
00196       for(i=k+1;i<Mat->NbColumns;++i) {
00197         
00198         /* pour tous les indices i > k */
00199         value_multiply(m1,Mat->p[j][i],Mat->p[k][k]);
00200         value_multiply(m2,Mat->p[j][k],Mat->p[k][i]);
00201         value_subtract(m3,m1,m2);
00202         value_division(Mat->p[j][i],m3,gcd);
00203       }
00204     }   
00205   }
00206 
00207   /* Clear all the 'Value' variables */
00208   value_clear(m1); value_clear(m2); 
00209   value_clear(m3); value_clear(gcd); value_clear(tmp);
00210   
00211   /* The matrix 'Mat' is full ranked, return 1 */
00212   return 1;
00213 } /* TestRank */

00576 /* D = Dimension of polytope (ray space)                                */
00577 /* L = Dimension of Lineality space (number of lines, bid)              */
00578 /* E = Dimension of Affine hull (number of equations)                   */
00579 /* n = number of data indices                                           */
00580 /* m = number of parameters                                             */
00581 /* full domain:                                                         */
00582 /*     n + m = D + E + L                                                */
00583 /* projected domains:                                                   */
00584 /*     m = D_m + E_m + L_m                                              */
00585 /*     n = D_n + E_n + L_n                                              */
00586 /* What dimension M-face, when projected onto parameter space,          */
00587 /* will give an L_m-face?                                               */
00588 /* What are the conditions?                                             */
00589 /*   - at least one vertex                                              */
00590 /*   - number of rays >= D_m+1 after removal of redundants              */
00591 /*                                                                      */
00592 /* dim of face    nb saturated constraints   nb saturated lines,rays    */
00593 /* -----------    ------------------------   -----------------------    */
00594 /* (0+L)-face     all E eqns + >=D ineq      all L lines + 1 ray        */
00595 /* (M+L)-face     all E eqns + >=(D-M) ineq  all L lines + >=(M+1) rays */
00596 /* (D+L)-face     all E eqns + 0 ineq        all L lines + >=(D+1) rays */
00597 /*----------------------------------------------------------------------*/
00598 /*----------------------------------------------------------------------*/
00599 /* scan_m_face                                                          */
00600 /*      pos : the next candidate constraint position                    */
00601 /*    nb_un : number of saturated constraints needed to finish a face   */
00602 /*        D : the source polyhedron (context included )                 */
00603 /*       mf : bit-array marking rays which are saturated so far         */
00604 /* From Global area:                                                    */
00605 /* ----------------                                                     */
00606 /*        n : number of data indices                                    */
00607 /*        m : number of parameters                                      */
00608 /*  egalite : boolean vector marking saturated constraints in m-face    */
00609 /*      Sat : Saturation Matrix (row=constraints, col=rays)             */
00610 /*       ws : working space size                                        */
00611 /*       nr : (NbRays-1)/32 + 1                                         */
00612 /*                                                                      */
00613 /* Recursive function to find the rays and vertices of each m-face      */
00614 /*----------------------------------------------------------------------*/
00615 static void scan_m_face(int pos,int nb_un,Polyhedron *D,unsigned int *mf) {
00616   /* pos   - the next candidate constraint position */
00617   /* nb_un - the number of constraints needed to finish a face */
00618   /* D     - the source polyhedron */
00619   /* mf    - (bit vector) marks rays that are saturated so far */
00620 
00621   unsigned int *new_mf;
00622 
00623 #ifdef DEBUGPP4
00624   fprintf(stderr,"Start scan_m_face(pos=%d, nb_un=%d, n=%d, m=%d\n",
00625           pos,nb_un,n,m);
00626   fprintf(stderr,"mf = ");
00627   {
00628     int i;
00629     for(i=0;i<nr;i++)
00630       fprintf(stderr,"%08X", mf[i]);
00631     fprintf(stderr,"\nequality = [");
00632     for(i=0;i<D->NbConstraints;i++)
00633       fprintf(stderr," %1d",egalite[i]);
00634     fprintf(stderr,"]\n");
00635   }
00636 #endif
00637 
00638   if(nb_un == 0) {      /* Base case */   
00639     int i;
00640     
00641     /*********** ELIMINATION OF REDUNDANT FACES ***********/
00642     /* if all these vertices also verify a previous constraint */
00643     /* which is NOT already selected, we eliminate this face */
00644     /* This keeps the lexicographically greatest selection */
00645         for(i=0;i<pos-1;i++)
00646         {
00647                 if(egalite[i])
00648                         continue;       /* already selected */
00649       
00650                 /* if Sat[i] & mf == mf then it's redundant */
00651 #ifdef DEBUGPP4
00652                 fprintf(stderr, "Sat[%d]\n", i);
00653 #endif
00654                 if (bit_vector_includes(Sat->p[i], nr, mf)) {
00655 #ifdef DEBUGPP4
00656                   fprintf(stderr, "Redundant with constraint %d\n", i);
00657 #endif
00658                   return;       /* it is redundant */
00659                 }
00660         }
00661     /********* END OF ELIMINATION OF DEGENERATE FACES *********/    
00662     /* Now check for other constraints that are verified */
00663     for (i = pos; i < D->NbConstraints; ++i) {
00664       if (bit_vector_includes(Sat->p[i], nr, mf))
00665         egalite[i] = 1;
00666     }
00667     /* if we haven't found a constraint verified by all */
00668     /* the rays, its OK, it's a new face. */
00669     traite_m_face(D, mf, egalite);
00670     for (i = pos; i < D->NbConstraints; ++i)
00671       egalite[i] = 0;
00672     return;
00673   }
00674 
00675   /* See if there are enough constraints left to finish */
00676   if((pos+nb_un)>D->NbConstraints) return;
00677   
00678   /* Recurring part of the procedure */
00679   /* Add the pos'th constraint, compute new saturation vector */
00680   {     
00681     int k;
00682     new_mf  = (unsigned int *)malloc(nr*sizeof(unsigned int));
00683     for (k=0; k<nr; k++)
00684       new_mf[k] = mf[k] & Sat->p[pos][k];
00685   }
00686 #ifdef DEBUGPP4
00687 fprintf(stderr,"new_mf = ");
00688  { 
00689    int i;
00690    for(i=0;i<nr;i++) {
00691      fprintf(stderr,"%08X", new_mf[i]);
00692    }
00693    fprintf(stderr,"\ncount(new_mf) = %d\n",count_sat(new_mf));
00694  }
00695 #endif
00696 
00697   {
00698         int c;
00699         c = count_sat(new_mf);
00700         /* optimization : at least m_dim+1 rays must be saturated to add this constraint */
00701         if (c>m_dim )
00702         {
00703                 int redundant = 0;
00704 
00705                 egalite[pos]=1;         /* Try it with the pos-th constraint */
00706 
00707                 /* If this constraint does not change anything,
00708                  * it is redundant with respect to the selected
00709                  * equalities and the remaining inequalities.
00710                  * Check whether it is redundant with respect
00711                  * to just the selected equalities.
00712                  */
00713                 if( c==count_sat(mf) ) {
00714                     int i, c, j;
00715 
00716                     for (i = 0, c = 0; i < D->NbConstraints; ++i) {
00717                         if (egalite[i] == 0 || egalite[i] == -1)
00718                             continue;
00719                         for (j = 0; j < D->Dimension+1; ++j)
00720                             value_assign(CTest->p[j][c], 
00721                                          D->Constraint[i][j+1]);
00722                         ++c;
00723                     }
00724                     CTest->NbColumns = c;
00725 #ifdef DEBUGPP41
00726                     Matrix_Print(stderr,P_VALUE_FMT,CTest);
00727 #endif
00728                     redundant = !TestRank(CTest);
00729                 }
00730 
00731                 /* Do not decrement nb_un if equality is redundant. */
00732                 if( redundant )
00733                 {
00734                    egalite[pos]=-1;     /* Don't use in further redundance test
00735                                          */
00736                    scan_m_face(pos+1,nb_un,D,new_mf);
00737                 }
00738                 else
00739                 {
00740                    scan_m_face(pos+1,nb_un-1,D,new_mf);
00741                 }
00742         }
00743   }
00744  free(new_mf);
00745  egalite[pos]=0;                /* Try it without the pos-th constraint */
00746  if ((pos+nb_un)>=D->NbConstraints) return;
00747  scan_m_face(pos+1,nb_un,D,mf);
00748  return;
00749 } /* scan_m_face */

00751 /*
00752  * Create a saturation matrix with rows correspond to the constraints and 
00753  * columns correspond to the rays of the polyhedron 'Pol'. Global variable 
00754  * 'nr' is set in the function.             
00755  */
00756 static SatMatrix *Poly2Sat(Polyhedron *Pol,unsigned int **L) { 
00757   
00758   SatMatrix *Sat;
00759   int i, j, k, kx;
00760   unsigned int *Temp;
00761   Value *p1, *p2, p3,tmp;
00762   unsigned Dimension, NbRay, NbCon, bx;
00763   
00764   /* Initialize all the 'Value' variables */
00765   value_init(p3); value_init(tmp);
00766   
00767   NbRay = Pol->NbRays;
00768   NbCon = Pol->NbConstraints;
00769   Dimension = Pol->Dimension+1;                /* Homogeneous Dimension */
00770   
00771   /* Build the Sat matrix */
00772   nr      = (NbRay - 1)/(sizeof(int)*8) + 1;   /* Set globally */
00773   Sat     = SMAlloc(NbCon,nr);
00774   Temp     = (unsigned int *)malloc(nr*sizeof(unsigned int));
00775   memset(Sat->p_init,0,nr*NbCon*sizeof(int));
00776   memset(Temp,0,nr*sizeof(unsigned int));
00777   kx=0; bx=MSB;
00778   for (k=0; k<NbRay; k++) { 
00779     for (i=0; i<NbCon; i++) {
00780       p1 = &Pol->Constraint[i][1];
00781       p2 = &Pol->Ray[k][1];
00782       value_set_si(p3,0);
00783       for (j=0;j<Dimension;j++) {
00784         value_multiply(tmp,*p1,*p2);
00785         value_addto(p3,p3,tmp);
00786         p1++; p2++;
00787       }
00788       if (value_zero_p(p3))
00789         Sat->p[i][kx]|=bx;
00790     }
00791     Temp[kx] |= bx;
00792     NEXT(kx, bx);
00793   }
00794   
00795   /* Set 'L' to an array containing ones in every bit position of its */
00796   /* elements.                                                        */
00797   *L = Temp;
00798   
00799   /* Clear all the 'Value' variables */
00800   value_clear(p3); value_clear(tmp);
00801   
00802   return Sat;
00803 } /* Poly2Sat */

00950 /* Elim_Columns                                                         */
00951 /* Eliminate columns from A, using the equalities in E.                 */
00952 /* ref and p are precalculated by PreElim_Columns, using E;             */
00953 /* these two vectors are used to construct the new constraint matrix    */
00954 /*----------------------------------------------------------------------*/
00955 Polyhedron *Elim_Columns(Polyhedron *A,Polyhedron *E,int *p,int *ref) {
00956   
00957   int i,l,c;
00958   Matrix *M, *C;
00959   Polyhedron *R;
00960   Value tmp1,tmp2;
00961   
00962   /* Initialize all the 'Value' variables */
00963   value_init(tmp1); value_init(tmp2);
00964   
00965 #ifdef DEBUGPP32
00966   fprintf(stderr,"\nElim_Columns starting\n");
00967   fprintf(stderr,"A =\n" );
00968   Polyhedron_Print(stderr,P_VALUE_FMT,A);
00969 #endif
00970   
00971   /* Builds M = constraint matrix of A, useless columns zeroed */
00972   M = Polyhedron2Constraints(A);
00973   for(l=0;l<E->NbEq;++l) {    
00974     for(c=0;c<M->NbRows;++c) {
00975       if(value_notzero_p(M->p[c][p[l]])) {
00976         
00977         /* A parameter to eliminate here ! */
00978         for(i=1;i<M->NbColumns;++i) {
00979           value_multiply(tmp1,M->p[c][i],E->Constraint[l][p[l]]);
00980           value_multiply(tmp2,E->Constraint[l][i],M->p[c][p[l]]);
00981           value_subtract(M->p[c][i],tmp1,tmp2);
00982         }
00983       }
00984     }
00985   }

00987 #ifdef DEBUGPP32
00988   fprintf(stderr,"\nElim_Columns after zeroing columns of A.\n");

00991 #endif
00992   
00993   /* Builds C = constraint matrix, useless columns eliminated */
00994   C = Matrix_Alloc(M->NbRows,M->NbColumns - E->NbEq);
00995   for(l=0;l<C->NbRows;++l)
00996     for(c=0;c<C->NbColumns;++c) {
00997       value_assign(C->p[l][c],M->p[l][ref[c]]);
00998     }
00999     
01000 #ifdef DEBUGPP32
01001   fprintf(stderr,"\nElim_Columns after eliminating columns of A.\n");
01002   fprintf(stderr,"C =\n");
01003   Matrix_Print(stderr,P_VALUE_FMT,C);
01004 #endif
01005     
01006   R = Constraints2Polyhedron(C,ws);
01007   Matrix_Free(C);
01008   Matrix_Free(M);
01009   value_clear(tmp1); value_clear(tmp2);

01011 } /* Elim_Columns */

01367

01370   Polyhedron *dx, *d1, *d2;
01371   Param_Domain *p1, *p2, *p2prev, *PDNew;

01391 #endif

01396      p1->F = (unsigned *) malloc (nv * sizeof(unsigned));

01442 #endif

01444 
01445 #ifdef DEBUGPP5
01446          fprintf(stderr,"Empty d1\n");
01447 #endif
01448          if (d1) 
01449            Domain_Free(d1);
01450          Domain_Free(dx);
01451          
01452          if (!d2 || emptyQ(d2) || d2->NbEq!=0) {
01453            
01454 #ifdef DEBUGPP5
01455            fprintf( stderr, "Empty d2 (deleting)\n");
01456 #endif
01457            /* dx = p1->Domain = p2->Domain */
01458            if (d2) Domain_Free(d2);
01459            
01460            /* Update p1 */
01461            for (i=0;i<nv;i++)
01462              p1->F[i] |= p2->F[i];

01465            p2prev->next = p2->next;

01475 #endif

01478              p1->F[i] |= p2->F[i];

01481            Domain_Free( p2->Domain );
01482            p2->Domain = d2;
01483          }
01484        }
01485        else { /* d1 is not empty */         
01486          if (!d2 || emptyQ(d2) || d2->NbEq!=0) {

01494            Domain_Free(dx);
01495

01533         Polyhedron *C = DomainConvex(p1->Domain, working_space);

01537    } /* end of p1 scan */

01893 /*
01894  * Scales the parametric polyhedron such that all vertices are integer.
01895  */
01896 void Param_Polyhedron_Scale_Integer(Param_Polyhedron *PP, Polyhedron **P,
01897                                     Value *det, unsigned MaxRays)
01898 {
01899   int i;
01900   int nb_param, nb_vars;
01901   Vector *denoms;
01902   Param_Vertices *V;
01903   Value global_var_lcm;
01904   Matrix *expansion;
01905 
01906   value_set_si(*det, 1);
01907   if (!PP->nbV)
01908     return;
01909 
01910   nb_param = PP->D->Domain->Dimension;
01911   nb_vars = PP->V->Vertex->NbRows;
01912 
01913   /* Scan the vertices and make an orthogonal expansion of the variable
01914      space */
01915   /* a- prepare the array of common denominators */
01916   denoms = Vector_Alloc(nb_vars);
01917   value_init(global_var_lcm);
01918 
01919   /* b- scan the vertices and compute the variables' global lcms */
01920   for (V = PP->V; V; V = V->next)
01921     for (i = 0; i < nb_vars; i++)
01922       value_lcm(denoms->p[i], denoms->p[i], V->Vertex->p[i][nb_param+1]);
01923 
01924   value_set_si(global_var_lcm, 1);
01925   for (i = 0; i < nb_vars; i++) {
01926     value_multiply(*det, *det, denoms->p[i]);
01927     value_lcm(global_var_lcm, global_var_lcm, denoms->p[i]);
01928   }
01929 
01930   /* scale vertices */
01931   for (V = PP->V; V; V = V->next)
01932     for (i = 0; i < nb_vars; i++) {
01933       Vector_Scale(V->Vertex->p[i], V->Vertex->p[i], denoms->p[i], nb_param+1);
01934       Vector_Normalize(V->Vertex->p[i], nb_param+2);
01935     }
01936 
01937   /* the expansion can be actually writen as global_var_lcm.L^{-1} */
01938   /* this is equivalent to multiply the rows of P by denoms_det */
01939   for (i = 0; i < nb_vars; i++)
01940     value_division(denoms->p[i], global_var_lcm, denoms->p[i]);
01941 
01942   /* OPT : we could use a vector instead of a diagonal matrix here (c- and d-).*/
01943   /* c- make the quick expansion matrix */
01944   expansion = Matrix_Alloc(nb_vars+nb_param+1, nb_vars+nb_param+1);
01945   for (i = 0; i < nb_vars; i++)
01946     value_assign(expansion->p[i][i], denoms->p[i]);
01947   for (i = nb_vars; i < nb_vars+nb_param+1; i++)
01948     value_assign(expansion->p[i][i], global_var_lcm);
01949 
01950   /* d- apply the variable expansion to the polyhedron */
01951   if (P)
01952     *P = Polyhedron_Preimage(*P, expansion, MaxRays);
01953 
01954   Matrix_Free(expansion);
01955   value_clear(global_var_lcm);
01956   Vector_Free(denoms);
01957 }
